    // 监听自动刷新
    initAutoRefreshListener() {
      const self = this;
      let lastClubCount = 0;
      let lastStorageCount = 0;

      // 方法1：拦截页面导航事件（切换页面时刷新）
      if (typeof UTGameFlowNavigationController !== 'undefined' && UTGameFlowNavigationController.prototype.pushViewController) {
        const originalPush = UTGameFlowNavigationController.prototype.pushViewController;
        UTGameFlowNavigationController.prototype.pushViewController = function(...args) {
          const result = originalPush.apply(this, args);
          setTimeout(() => self.autoRefreshPanel(), 500);
          return result;
        };
      }

      // 方法2：监听 repositories.Item club 缓存重置
      if (repositories.Item && repositories.Item.club && repositories.Item.club.items) {
        try {
          const originalReset = repositories.Item.club.items.reset;
          if (originalReset) {
            repositories.Item.club.items.reset = function(...args) {
              const result = originalReset.apply(this, args);
              setTimeout(() => self.autoRefreshPanel(), 100);
              return result;
            };
          }
        } catch (e) {}
      }

      // 方法3：监听 repositories.Item storage 缓存重置（尝试，不一定触发）
      if (repositories.Item && repositories.Item.storage && repositories.Item.storage.reset) {
        try {
          const originalReset = repositories.Item.storage.reset;
          repositories.Item.storage.reset = function(...args) {
            const result = originalReset.apply(this, args);
            setTimeout(() => self.autoRefreshPanel(), 100);
            return result;
          };
        } catch (e) {}
      }

      // 方法4：定时检查球员数量变化（兜底方案）
      let initialized = false;
      setInterval(() => {
        try {
          const clubCount = repositories.Item?.club?.items?.length || 0;
          const storageCount = repositories.Item?.storage?.length || 0;
          if (initialized && (clubCount !== lastClubCount || storageCount !== lastStorageCount)) {
            console.log(`[EAFC Grinder] 检测到球员变化 (俱乐部: ${lastClubCount} → ${clubCount}, 仓库: ${lastStorageCount} → ${storageCount})，刷新面板...`);
            self.autoRefreshPanel();
          }
          lastClubCount = clubCount;
          lastStorageCount = storageCount;
          initialized = true;
        } catch (e) {}
      }, 3000);

      console.log("[EAFC Grinder] 自动刷新监听已注册");
    },

    // 自动刷新面板（如果面板存在）
    autoRefreshPanel() {
      const existing = document.querySelector(".grinder-panel");
      if (existing) {
        console.log("[EAFC Grinder] 正在刷新面板...");
        this.renderPanel();
      }
    },

    // 分页加载所有球员（俱乐部 + 重复仓库）
    async loadAllPlayers() {
      try {
        const current = _appMain?.getRootViewController?.();
        if (!current) {
          console.error("[EAFC Grinder] 无法获取根视图控制器");
          return;
        }

        // 获取俱乐部球员总数
        const stats = await new Promise((resolve, reject) => {
          services.Club.getStats().observe(current, (obs, result) => {
            obs.unobserve(current);
            if (result.success) {
              resolve(result.response);
            } else {
              reject(new Error("获取俱乐部统计失败"));
            }
          });
        });

        const playerStats = stats.stats?.find((s) => s.type === "players");
        if (!playerStats) {
          console.error("[EAFC Grinder] 未找到球员统计数据");
          return;
        }

        const totalPlayers = playerStats.count;
        console.log(`[EAFC Grinder] 俱乐部共有 ${totalPlayers} 名球员，开始全量加载...`);

        // 创建加载遮罩
        const overlay = document.createElement("div");
        overlay.className = "grinder-loading-overlay";
        overlay.innerHTML = `
          <div class="loading-text">正在加载球员数据...</div>
          <div class="loading-bar-bg"><div class="loading-bar-fill"></div></div>
          <div class="loading-detail">0%</div>
        `;
        document.body.appendChild(overlay);
        const barFill = overlay.querySelector(".loading-bar-fill");
        const detailEl = overlay.querySelector(".loading-detail");
        let pagesLoaded = 0;

        const updateProgress = () => {
          pagesLoaded++;
          const pct = Math.min(100, Math.round(pagesLoaded / totalPages * 100));
          barFill.style.width = pct + "%";
          detailEl.textContent = pct + "%";
        };

        // 分页加载俱乐部球员，每页200人，并发
        const batchSize = 200;
        const totalPages = Math.ceil(totalPlayers / batchSize);
        const CONCURRENCY = 8;

        const fetchPage = (page) => new Promise((resolve, reject) => {
          const criteria = new UTSearchCriteriaDTO();
          criteria.type = "player";
          criteria.sortBy = "ovr";
          criteria.sort = "desc";
          criteria.count = batchSize;
          criteria.offset = page * batchSize;

          services.Club.search(criteria).observe(current, (obs, res) => {
            obs.unobserve(current);
            if (res.success && JSUtils.isObject(res.response)) {
              console.log(`[EAFC Grinder] 已加载俱乐部第 ${page + 1}/${totalPages} 页 (${res.response.items?.length || 0} 名球员)`);
              updateProgress();
              resolve(res.response);
            } else {
              reject(new Error("搜索失败"));
            }
          });
        });

        for (let i = 0; i < totalPages; i += CONCURRENCY) {
          const pageBatch = [];
          for (let j = i; j < Math.min(i + CONCURRENCY, totalPages); j++) {
            pageBatch.push(fetchPage(j));
          }
          await Promise.all(pageBatch);
        }

        console.log(`[EAFC Grinder] 俱乐部加载完成，共 ${totalPlayers} 名球员`);

        // 加载重复仓库（storage pile）
        await this.loadStoragePlayers();

        // 移除遮罩
        overlay.remove();

        this.showPlayerPanel();
        this.initSBCFillButton();
        this.initPriceCache();
        this.initClubPriceDisplay();

      } catch (error) {
        const overlay = document.querySelector(".grinder-loading-overlay");
        if (overlay) overlay.remove();
        console.error("[EAFC Grinder] 加载球员时出错:", error);
      }
    },

    // 加载重复仓库球员（初始化时强制 EA 加载 storage 数据）
    async loadStoragePlayers() {
      try {
        const current = _appMain?.getRootViewController?.();
        if (!current) return;

        console.log("[EAFC Grinder] 正在加载重复仓库...");

        // 使用 services.Item.searchStorageItems 强制 EA 加载 storage 数据到缓存
        await new Promise((resolve, reject) => {
          services.Item.searchStorageItems(new UTSearchCriteriaDTO()).observe(current, (obs, res) => {
            obs.unobserve(current);
            if (res.success && JSUtils.isObject(res.response)) {
              resolve(res.response);
            } else {
              reject(new Error("搜索失败"));
            }
          });
        });

        const storagePlayers = this.getStoragePlayers();
        console.log(`[EAFC Grinder] 重复仓库加载完成 (${storagePlayers.length} 名球员)`);
      } catch (error) {
        console.warn("[EAFC Grinder] 加载重复仓库时出错:", error);
      }
    },

    // 通用：从 repository 读取球员
    _parseRepositoryPlayers(repository, isStorage = false) {
      const players = [];
      if (!repository) return players;

      const items = repository.items || repository;
      const coll = items._collection;

      if (coll && coll.length !== undefined) {
        for (let i = 0; i < coll.length; i++) {
          const item = coll[i];
          if (item && item.isPlayer && item.isPlayer() && item.loans == -1) {
            players.push(this._parsePlayerItem(item, isStorage));
          }
        }
        if (players.length > 0) return players;
      }

      if (items && typeof items.values === 'function') {
        const collection = items.values();
        if (collection && collection.length !== undefined) {
          for (let i = 0; i < collection.length; i++) {
            const item = collection[i];
            if (item && item.isPlayer && item.isPlayer() && item.loans == -1) {
              players.push(this._parsePlayerItem(item, isStorage));
            }
          }
          if (players.length > 0) return players;
        }
      }

      return players;
    },

    // 获取俱乐部球员列表
    getClubPlayers() {
      const players = this._parseRepositoryPlayers(repositories.Item.club);
      players.sort((a, b) => b.rating - a.rating);
      return players;
    },

    // 获取重复仓库球员列表
    getStoragePlayers() {
      const players = this._parseRepositoryPlayers(repositories.Item.storage, true);
      players.sort((a, b) => b.rating - a.rating);
      return players;
    },

    // 判断是否为租借球员（包括限时租借）
    isLoanPlayer(p) {
      const item = p._item;
      if (!item) return false;
      // loans 不等于 -1 表示按次数的租借
      if (p.loans != null && p.loans !== -1) return true;
      // endTime 不是 -1 表示限时租借
      if (item.endTime != -1) return true;
      return false;
    },

    // 判断是否为进化卡
    isUpgradedCard(p) {
      const item = p._item;
      if (!item) return false;
      try {
        const upgrades = typeof item.upgrades === 'string' ? JSON.parse(item.upgrades) : item.upgrades;
        return upgrades && (upgrades.enrolled !== undefined || upgrades.hasStatUpgrade !== undefined);
      } catch (e) {
        return false;
      }
    },

    // 解析球员项为统一格式
    _parsePlayerItem(item, isStorage = false) {
      let name = "Unknown";
      let staticData = null;

      try {
        staticData = item.getStaticData ? item.getStaticData() : (item._staticData || null);
        if (staticData) {
          name = staticData.name || item._commonName || "Unknown";
        } else if (item.definitionId) {
          staticData = repositories.Item.getStaticDataByDefId(item.definitionId);
          if (staticData) {
            name = staticData.name || "Unknown";
          }
        }
      } catch (e) {
        name = item._commonName || "Unknown";
      }

      return {
        id: item.id,
        definitionId: item.definitionId,
        duplicateId: item.duplicateId,
        name: name,
        firstName: staticData?.firstName || "",
        lastName: staticData?.lastName || "",
        commonName: item._commonName || "",
        rating: item.rating || item.overallRating || 0,
        position: item.prefPos || item.position || "",
        positions: item.basePossiblePositions || item.possiblePositions || [],
        rareflag: item.rareflag != null ? item.rareflag : (item._rareflag || 0),
        rarityName: this.getRarityName(item.rareflag != null ? item.rareflag : (item._rareflag || 0)).name,
        leagueId: item.leagueId || 0,
        leagueName: this.getLeagueName(item.leagueId),
        nationId: item.nationId || 0,
        nationName: this.getNationName(item.nationId),
        teamId: item.teamId || 0,
        clubName: staticData?.clubName || "",
        cardAssetId: item.cardAssetId || 0,

        weakFoot: item.weakFootAbility || 0,
        skillMoves: item.skillMoves || 0,
        height: item.height || 0,

        birthday: staticData?.birthday || 0,
        traits: item.traits || [],
        groups: item.groups || [],
        playStyles: item.playStyles || [],
        acceleRate: item.acceleRate || 0,

        timestamp: Date.now(),

      };
    },

    // 统计重复球员（definitionId 出现次数 > 1）
    getDuplicateStats() {
      const clubPlayers = this.getClubPlayers();
      const defIdCount = {};
      clubPlayers.forEach(p => {
        if (!defIdCount[p.definitionId]) {
          defIdCount[p.definitionId] = { count: 0, players: [] };
        }
        defIdCount[p.definitionId].count++;
        defIdCount[p.definitionId].players.push(p);
      });

      // 只保留重复的
      const duplicates = Object.entries(defIdCount)
        .filter(([_, v]) => v.count > 1)
        .map(([defId, v]) => ({
          definitionId: parseInt(defId),
          count: v.count,
          players: v.players,
        }))
        .sort((a, b) => b.count - a.count);

      return duplicates;
    },

    makeDraggable(element) {
      let isDragging = false;
      let startX, startY, initialRight, initialTop;

      const handle = element.querySelector(".grinder-drag-handle") || element;

      const onMouseDown = (e) => {
        // 如果点击的是按钮，不拖拽
        if (e.target.classList.contains("grinder-toggle-btn") ||
            e.target.tagName === "BUTTON") {
          return;
        }

        isDragging = true;
        startX = e.clientX;
        startY = e.clientY;

        const rect = element.getBoundingClientRect();
        initialRight = window.innerWidth - rect.right;
        initialTop = rect.top;

        element.style.transition = "none";
        element.style.cursor = "grabbing";
        e.preventDefault();
      };

      const onMouseMove = (e) => {
        if (!isDragging) return;

        const deltaX = e.clientX - startX;
        const deltaY = e.clientY - startY;

        const newRight = initialRight - deltaX;
        const newTop = initialTop + deltaY;

        // 确保面板在可视区域内
        const maxRight = window.innerWidth - element.offsetWidth;
        const maxY = window.innerHeight - element.offsetHeight;

        const finalRight = Math.max(0, Math.min(newRight, maxRight));
        const finalTop = Math.max(0, Math.min(newTop, maxY));

        element.style.right = finalRight + "px";
        element.style.top = finalTop + "px";
        element.style.left = "auto";

        // 保存位置
        this.panelPosition.right = finalRight;
        this.panelPosition.top = finalTop;
      };

      const onMouseUp = () => {
        isDragging = false;
        element.style.transition = "";
        element.style.cursor = "";
      };

      // 鼠标事件
      handle.addEventListener("mousedown", onMouseDown);
      document.addEventListener("mousemove", onMouseMove);
      document.addEventListener("mouseup", onMouseUp);

      // 触摸事件（移动端支持）
      handle.addEventListener("touchstart", (e) => {
        if (e.target.tagName === "BUTTON") return;
        const touch = e.touches[0];
        onMouseDown({ clientX: touch.clientX, clientY: touch.clientY, target: e.target, preventDefault: () => e.preventDefault() });
      });

      document.addEventListener("touchmove", (e) => {
        const touch = e.touches[0];
        onMouseMove({ clientX: touch.clientX, clientY: touch.clientY });
      });

      document.addEventListener("touchend", onMouseUp);
    },

    // 从 futbin 获取评分→市场价格映射，每 30 分钟刷新一次
    fetchPriceMap() {
      const self = this;

      const doFetch = () => {
        let platform = "ps";
        try {
          const user = services.User.getUser().getSelectedPersona();
          if (!user.isXbox && !user.isPlaystation && !user.isStadia) {
            platform = "pc";
          }
        } catch (e) {}

        const year = typeof APP_YEAR_SHORT !== "undefined" ? APP_YEAR_SHORT : "25";
        const url = "https://www.futbin.org/futbin/api/" + year + "/getSTCCheapest?platform=" + (platform === "pc" ? "PC" : "PS");

        GM_xmlhttpRequest({
          method: "GET",
          url: url,
          headers: {
            "Content-type": "application/json",
            "Cache-Control": "no-cache",
            "Pragma": "no-cache"
          },
          onload(res) {
            try {
              const data = JSON.parse(res.response);
              const priceJson = {};
              let lowRating = 99;
              let baseLowPrice = 0;
              let highRating = 0;
              let baseHighPrice = 0;

              Object.values(data.data.players || {}).forEach(i => {
                if (i.data.players && i.data.players.length) {
                  const rating = i.data.data.rating;
                  const firstFive = i.data.players.slice(0, 5).map(p => p.LCPrice);
                  const avg = firstFive.reduce((a, b) => a + b, 0) / firstFive.length;
                  const price = Math.max(Math.round(avg), 600);
                  if (rating < lowRating) { lowRating = rating; baseLowPrice = price; }
                  if (rating > highRating) { highRating = rating; baseHighPrice = price; }
                  priceJson[rating] = price;
                }
              });

              // 线性插值填充金卡低段（75 ~ lowRating-1）
              const distributeValues = (startKey, endKey, startValue, endValue) => {
                const keyCount = endKey - startKey + 1;
                const totalDiff = endValue - startValue;
                const step = Math.floor(totalDiff / keyCount);
                const remainder = totalDiff % keyCount;
                for (let k = startKey, idx = 0; k <= endKey; k++, idx++) {
                  priceJson[k] = startValue + step * idx + (idx < remainder ? 1 : 0);
                }
              };

              // 75~81 如果 futbin 没有数据统一 350，有数据则线性插值填补空白
              for (let r = 75; r <= 81; r++) {
                if (!priceJson[r]) priceJson[r] = 350;
              }
              if (lowRating > 75) {
                distributeValues(75, lowRating - 1, 350, priceJson[lowRating] || 350);
              }

              // 银卡、铜卡统一 200
              for (let r = 45; r <= 74; r++) {
                priceJson[r] = 200;
              }

              // 高评分段（highRating+1 ~ 99）用 1.2 倍递增估算
              for (let r = highRating + 1; r <= 99; r++) {
                baseHighPrice = Math.round(baseHighPrice * 1.2);
                priceJson[r] = baseHighPrice;
              }

              self.priceMap = priceJson;
              console.log("[EAFC Grinder] 价格映射已更新, 共", Object.keys(priceJson).length, "个评分");
              self.renderPanel();
            } catch (e) {
              console.warn("[EAFC Grinder] 解析价格数据失败:", e);
            }
          },
          onerror() {
            console.warn("[EAFC Grinder] 获取价格数据失败（网络错误）");
          }
        });
      };

      // 立即执行一次，然后每 30 分钟刷新
      doFetch();
      if (this._priceMapTimer) clearInterval(this._priceMapTimer);
      this._priceMapTimer = setInterval(doFetch, 30 * 60 * 1000);
    },

      const self = this;

    // 从后端全量接口加载所有价格到 priceCacheMap
    _loadAllPrices() {
      const self = this;
      const platform = this._getPlatform();
      const url = this._getAllPricesUrl() + "?platform=" + platform;
      console.log("[EAFC Grinder] 开始加载全量价格数据...");
      const reqStart = performance.now();

      GM_xmlhttpRequest({
        method: "GET",
        url: url,
        headers: { "ngrok-skip-browser-warning": "true" },
        onload(res) {
          const reqTime = Math.round(performance.now() - reqStart);
          if (res.status === 503) {
            console.warn(`[EAFC Grinder] 全量价格接口返回 503（服务器预热中），${reqTime}ms`);
            return;
          }
          try {
            const data = JSON.parse(res.response);
            if (data.prices) {
              const count = Object.keys(data.prices).length;
              Object.entries(data.prices).forEach(([id, price]) => {
                self.priceCacheMap[Number(id)] = (price && price > 0) ? price : 0;
              });
              console.log(`[EAFC Grinder] 全量价格加载完成: ${count} 张卡, ${reqTime}ms`);

              self.renderPanel();
            }
          } catch (e) {
            console.error("[EAFC Grinder] 全量价格加载失败:", e);
          }
        },
        onerror() {
          console.error("[EAFC Grinder] 全量价格请求失败");
        }
      });
    },

    // 未分配球员价格显示 - 在 EA 原生未分配列表中注入价格标签
    initUnassignedPriceDisplay() {
      const self = this;
      const PRICE_LABEL_CLASS = "grinder-unassigned-price";

      // 添加价格标签样式
      const style = document.createElement("style");
      style.textContent = `
        .${PRICE_LABEL_CLASS} {

      `;

      // 从未分配列表中提取球员数据
      function getUnassignedPlayers() {
        const players = [];
        try {
          const unassignedRepo = repositories?.Item?.unassigned;
          if (!unassignedRepo) return players;

          const iterator = unassignedRepo.values();
          if (!iterator) return players;

          for (let i = 0; i < iterator.length; i++) {
            const item = iterator[i];
            if (item && item.isPlayer && item.isPlayer() && item.loans == -1) {
              players.push(item);
            }
          }
        } catch (e) {
          console.warn("[EAFC Grinder] 获取未分配球员失败:", e);
        }
        return players;
      }

      // 为未分配球员批量查价
      async function fetchUnassignedPrices(unassignedItems, needPrice) {
        isFetching = true;

        const platform = self._getPlatform();
        const defIds = needPrice.map(p => Number(p.definitionId));
        needPrice.forEach(p => self._fetchingDefIds.add(Number(p.definitionId)));

        // 构建 players 对象（futbin 兜底需要球员属性）
        const players = {};
        needPrice.forEach(p => {
          players[p.definitionId] = {
            nationId: p.nationId,
            leagueId: p.leagueId,
            teamId: p.teamId,
            rating: p.rating
          };
        });

        const reqStart = performance.now();
        console.log(`[EAFC Grinder] 发送未分配查价请求: ${defIds.length} 个ID...`);

        await new Promise((resolve) => {
          GM_xmlhttpRequest({
            method: "POST",
            url: self._getPriceServerUrl(),
            headers: { "Content-Type": "application/json" },
            data: JSON.stringify({ definitionIds: defIds, platform: platform, players: players }),
            onload(res) {
              const reqTime = Math.round(performance.now() - reqStart);
              console.log(`[EAFC Grinder] 未分配查价响应 (${reqTime}ms), status: ${res.status}`);
              try {
                const data = JSON.parse(res.response);
                if (data.prices) {
                  Object.entries(data.prices).forEach(([defId, result]) => {
                    const price = result.price;
                    self.priceCacheMap[Number(defId)] = (price && price > 0) ? price : 0;
                    self._fetchingDefIds.delete(Number(defId));
                  });
                }
              } catch (e) {
                console.error("[EAFC Grinder] 未分配查价失败:", e);
                defIds.forEach(id => self._fetchingDefIds.delete(id));
              }
              resolve();
            },
            onerror() {
              console.error("[EAFC Grinder] 未分配查价请求失败");
              defIds.forEach(id => self._fetchingDefIds.delete(id));
              resolve();
            }
          });
        });

        console.log("[EAFC Grinder] 未分配查价完成，更新DOM...");
        injectPricesToDOM(unassignedItems);
        isFetching = false;
      }

      // 在 DOM 中注入价格标签 - 通过 rating+name+nationId+leagueId+teamId+6stats 匹配
      function injectPricesToDOM(unassignedItems) {
        // 构建 item 查找表: key = "rating|name|nationId|leagueId|teamId|pac|sho|pas|dri|def|phy" -> [items]
        const itemMap = {};
        unassignedItems.forEach(item => {
          const knownAs = item._staticData?.knownAs || '';
          const lastName = item._staticData?.lastName || item.lastName || '';
          const name = (knownAs && knownAs !== '---') ? knownAs : lastName;
          const attrs = item.attributes || [];
          const key = `${item.rating}|${name}|${item.nationId}|${item.leagueId}|${item.teamId}|${attrs[0]||''}|${attrs[1]||''}|${attrs[2]||''}|${attrs[3]||''}|${attrs[4]||''}|${attrs[5]||''}`;
          if (!itemMap[key]) itemMap[key] = [];
          itemMap[key].push(item);
        });

        // 从 DOM 元素提取匹配数据
        function extractKeyFromLi(li) {
          const rating = li.querySelector('.rating')?.textContent?.trim() || '';
          const name = li.querySelector('.name')?.textContent?.trim() || '';
          // 从图片 URL 提取 ID: flags/dark/14.png -> 14
          const nationSrc = li.querySelector('img[src*="flags"]')?.getAttribute('src') || '';
          const leagueSrc = li.querySelector('img[src*="leagues"]')?.getAttribute('src') || '';
          const clubSrc = li.querySelector('img[src*="clubs"]')?.getAttribute('src') || '';
          const nationId = nationSrc.match(/(\d+)\.png/)?.[1] || '';
          const leagueId = leagueSrc.match(/(\d+)\.png/)?.[1] || '';
          const teamId = clubSrc.match(/(\d+)\.png/)?.[1] || '';
          // 6项小数据，按顺序取值（守门员标签不同，不能按名称匹配）
          const statSpans = li.querySelectorAll('.player-stats-data-component li');
          const statValues = Array.from(statSpans).slice(0, 6).map(s => {
            const v = s.querySelector('.value');
            return v ? v.textContent.trim() : '';
          });
          while (statValues.length < 6) statValues.push('');
          return `${rating}|${name}|${nationId}|${leagueId}|${teamId}|${statValues.join('|')}`;
        }

        // 遍历所有 section 的 li 注入价格
        const sections = document.querySelectorAll(".ut-sectioned-item-list-view");
        let matched = 0, unmatched = 0;
        sections.forEach(section => {
          const playerItems = section.querySelectorAll("li.listFUTItem:not(.has-action)");
          playerItems.forEach(li => {
            // 跳过非球员物品（徽章、球衣等没有 .player 类）
            if (!li.querySelector(".player.item")) return;
            const key = extractKeyFromLi(li);
            const items = itemMap[key];
            const item = items && items.length > 0 ? items.shift() : null;

            if (!item) {
              unmatched++;
              return;
            }
            matched++;

            const definitionId = Number(item.definitionId);

            let priceLabel = li.querySelector(`.${PRICE_LABEL_CLASS}`);
            if (!priceLabel) {
              priceLabel = document.createElement("div");
              priceLabel.className = PRICE_LABEL_CLASS;
              li.appendChild(priceLabel);
            }

            const price = self.priceCacheMap[definitionId];
            if (price && price > 0) {
              priceLabel.textContent = price.toLocaleString();
              priceLabel.classList.remove("no-price");
            } else {
              priceLabel.textContent = "N/A";
              priceLabel.classList.add("no-price");
            }
          });
        });

        console.log(`[EAFC Grinder] 未分配列表价格渲染完成 - 匹配: ${matched}, 未匹配: ${unmatched}`);
      }

      // 定时检测：页面有未分配列表且未渲染价格时，触发查价和渲染
      let isFetching = false;
      setInterval(() => {
        const sections = document.querySelectorAll(".ut-sectioned-item-list-view");
        if (sections.length === 0) return;

        // 检查是否还有没渲染价格的球员卡（跳过非球员物品）
        let needsRender = false;
        sections.forEach(section => {
          const items = section.querySelectorAll("li.listFUTItem:not(.has-action)");
          items.forEach(li => {
            if (!li.querySelector(".player.item")) return;
            if (!li.querySelector(`.${PRICE_LABEL_CLASS}`)) {
              needsRender = true;
            }
          });
        });

        if (!needsRender || isFetching) return;

        const unassignedItems = getUnassignedPlayers();
        if (unassignedItems.length === 0) return;

        // 去重
        const seenDefIds = new Set();
        const uniqueItems = [];
        unassignedItems.forEach(item => {
          const defId = Number(item.definitionId);
          if (!seenDefIds.has(defId)) {
            seenDefIds.add(defId);
            uniqueItems.push(item);
          }
        });

        // 过滤出需要查价的球员
        const needPrice = uniqueItems.filter(item => {
          const defId = Number(item.definitionId);
          return !(defId in self.priceCacheMap) && !self._fetchingDefIds.has(defId);
        });

        if (needPrice.length > 0) {
          fetchUnassignedPrices(unassignedItems, needPrice);
        } else {
          injectPricesToDOM(unassignedItems);
        }
      }, 500);
    },

    // SBC 一键填充功能
    fetchPricesOnDemand(pagePlayers) {
      const self = this;
      const missing = pagePlayers.filter(p => {
        if (p.definitionId in this.priceCacheMap) return false;
        if (this._onDemandFetching && this._onDemandFetching.has(p.definitionId)) return false;
        return true;
      });
      if (missing.length === 0) return;

      if (!this._onDemandFetching) this._onDemandFetching = new Set();
      missing.forEach(p => this._onDemandFetching.add(p.definitionId));

      const defIds = missing.map(p => p.definitionId);
      const platform = this._getPlatform();

      // 构建 players 对象（futbin 兜底需要球员属性）
      const players = {};
      missing.forEach(p => {
        players[p.definitionId] = {
          nationId: p.nationId,
          leagueId: p.leagueId,
          teamId: p.teamId,
          rating: p.rating
        };
      });

      GM_xmlhttpRequest({
        method: "POST",
        url: self._getPriceServerUrl(),
        headers: { "Content-Type": "application/json" },
        data: JSON.stringify({ definitionIds: defIds, platform: platform, players: players }),
        onload(res) {
          try {
            const data = JSON.parse(res.response);
            if (data.prices) {
              Object.entries(data.prices).forEach(([defId, result]) => {
                const price = result.price;
                self.priceCacheMap[defId] = (price && price > 0) ? price : 0;
                self._onDemandFetching.delete(defId);
              });
            }
          } catch (e) {
            console.error("[EAFC Grinder] 按需查价失败:", e);
            missing.forEach(p => {
              self._onDemandFetching.delete(p.definitionId);
              self.priceCacheMap[p.definitionId] = 0;
            });
          }
          self.renderPanel();
        },
        onerror() {
          console.error("[EAFC Grinder] 按需查价请求失败");
          missing.forEach(p => {
            self._onDemandFetching.delete(p.definitionId);
            self.priceCacheMap[p.definitionId] = 0;
          });
          self.renderPanel();
        }
      });
    },

    // 监听挑战切换
    _observeChallengeChange(controller) {
      const self = this;

      // 使用 MutationObserver 检测挑战名称变化
      let lastChallengeName = controller._challenge?.name || '';

      const checkInterval = setInterval(() => {
        if (!controller._challenge) return;

        const currentName = controller._challenge.name || '';
        if (currentName !== lastChallengeName) {

          self.currentChallenge = controller._challenge;
          // 更新按钮绑定的控制器引用
          self.currentSBCController = controller;
        }
      }, 1000);

      this._challengeCheckInterval = checkInterval;
    },

    async fillSBCSquad(controller, fillBtn) {
      // 从 repository 获取最新需求
      const challenge = controller._challenge;
      if (!challenge) {
        alert("❌ 无法获取 SBC 挑战信息");
        return;
      }

      // 尝试从 repository 获取完整需求
      let requirements = null;
      try {
        const repository = services.SBC.repository;
        const set = repository.getSetById(challenge.setId);
        if (set && set.challenges) {
          const currentChallenge = set.challenges.find(c => c.challengeId === challenge.challengeId || c.id === challenge.challengeId);
          if (currentChallenge && currentChallenge.elgReq) {
            requirements = currentChallenge.elgReq;
          }
        }
      } catch (e) {
        console.warn("[EAFC Grinder] 从 repository 获取需求失败:", e);
      }

      // fallback 使用 controller 的数据
      if (!requirements || requirements.length === 0) {
        requirements = challenge.elgReq || challenge.eligibilityRequirements || [];
      }

      if (requirements.length === 0) {
        alert("❌ 无法获取 SBC 需求信息");
        return;
      }

      fillBtn.disabled = true;
      fillBtn.textContent = "⏳ 正在填充...";
      fillBtn.style.opacity = "0.6";

      try {
        console.log("[EAFC Grinder] 开始分析 SBC 需求...");

        // 解析 SBC 需求
        const squadRequirements = this.parseSBCRequirements(requirements);

        // 从 formation 计算实际需要填充的球员数（总槽位 - 锁定槽位）
        try {
          const formation = controller._challenge.formation;
          const formationData = repositories.Squad.getFormation(formation);
          const brickIndices = controller._squad.getAllBrickIndices ? controller._squad.getAllBrickIndices() : [];
          squadRequirements.minPlayers = formationData.generalPositions.length - brickIndices.length;
          console.log("[EAFC Grinder] formation generalPositions:", formationData.generalPositions);
          console.log("[EAFC Grinder] challenge metadata:", controller._challenge);
        } catch(e) {}

        // === 统一走化学感知求解器（确保硬约束如 nationCountMax 等在选人阶段被遵守） ===
        if (true || squadRequirements.minChemistry > 0) {
          console.log("[EAFC Grinder] >>> 化学感知求解器分支");

          // 显示加载遮盖（与球员数据加载风格一致）
          const overlay = document.createElement('div');
          overlay.className = 'grinder-loading-overlay';
          overlay.innerHTML = `
            <div class="loading-text">正在努力填充中...</div>
            <div class="loading-bar-bg"><div class="loading-bar-fill" style="width:0%"></div></div>
            <div class="loading-detail">化学感知求解器运行中</div>
          `;
          document.body.appendChild(overlay);
          const barFill = overlay.querySelector('.loading-bar-fill');
          const detailEl = overlay.querySelector('.loading-detail');

          // 进度回调
          this._chemProgress = (pct, detail) => {
            try {
              barFill.style.width = pct + '%';
              if (detail) detailEl.textContent = detail;
            } catch(e) {}
          };

          const removeOverlay = () => {
            delete this._chemProgress;
            const el = document.querySelector('.grinder-loading-overlay');
            if (el) el.remove();
          };

          try {
            const eligiblePlayers = this.filterEligiblePlayers(squadRequirements);

            const optimalSquad = await this.selectOptimalSquadWithChemistry(eligiblePlayers, squadRequirements, null, controller);

            // ranges 可以重叠（如金卡法甲球员同时满足法甲和金卡两个范围），
            // 所以总需求人数就是 minPlayers，不是 range 的 count 简单相加
            const requiredCount = squadRequirements.minPlayers || 11;

            if (optimalSquad.length < requiredCount) {
              removeOverlay();
              alert("❌ 俱乐部中没有足够的球员满足 SBC 需求\n\n需要: " + requiredCount + " 名\n可用: " + optimalSquad.length + " 名");
              fillBtn.disabled = false;
              fillBtn.textContent = "⚡ 一键填充最优阵容";
              fillBtn.style.opacity = "1";
              return;
            }

            // 填充到阵容
            const fillLog = await this.fillSquadWithPlayers(controller, optimalSquad);

            removeOverlay();
            fillBtn.disabled = false;
            fillBtn.textContent = "⚡ 一键填充最优阵容";
            fillBtn.style.opacity = "1";
            return;
          } catch (err) {
            removeOverlay();
            throw err;
          }
        }

        // === TEAM_RATING 处理（无化学要求时） ===
        let fillRatings = null;
        if (squadRequirements.minTeamRating > 0) {
          try {
            fillRatings = this.calculateFillRatings(
              squadRequirements.minTeamRating,
              controller,
              squadRequirements.rarityGroups,
              squadRequirements.ratingRanges
            );
            console.log("[EAFC Grinder] TEAM_RATING 计算结果:", fillRatings);
          } catch (err) {
            alert("一键填充失败: " + err.message);
            fillBtn.disabled = false;
            fillBtn.textContent = "⚡ 一键填充最优阵容";
            fillBtn.style.opacity = "1";
            return;
          }
        }

        // 从球员数据库中筛选符合条件的球员
        const eligiblePlayers = this.filterEligiblePlayers(squadRequirements);

        // 选择最优阵容（评分最低但符合条件，若存在 TEAM_RATING 则按 fillRatings 指导）
        console.log("[EAFC Grinder] >>> 快速填充分支");
        const optimalSquad = this.selectOptimalSquad(eligiblePlayers, squadRequirements, fillRatings);

        // ranges 可以重叠，总需求人数就是 minPlayers
        const requiredCount = squadRequirements.minPlayers || 11;

        if (optimalSquad.length < requiredCount) {
          alert("❌ 俱乐部中没有足够的球员满足 SBC 需求\n\n需要: " + requiredCount + " 名\n可用: " + optimalSquad.length + " 名");
          fillBtn.disabled = false;
          fillBtn.textContent = "⚡ 一键填充最优阵容";
          fillBtn.style.opacity = "1";
          return;
        }

        // 填充到阵容
        const fillLog = await this.fillSquadWithPlayers(controller, optimalSquad);

        // 验证阵容评分
        if (squadRequirements.minTeamRating > 0) {
          const finalRatings = [];
          const fps = controller._squad.getFieldPlayers ? controller._squad.getFieldPlayers() : [];
          fps.forEach(fp => {
            if (fp.item && fp.item.rating) {
              finalRatings.push(fp.item.rating);
            }
          });
          const actualRating = this.calculateTeamRating(finalRatings);
          if (actualRating < squadRequirements.minTeamRating) {
            alert("警告: 填充后阵容评分为 " + actualRating + "，未达到要求的 " + squadRequirements.minTeamRating);
          }
        }

        // 直接恢复按钮状态
        fillBtn.disabled = false;
        fillBtn.textContent = "⚡ 一键填充最优阵容";
        fillBtn.style.opacity = "1";

      } catch (error) {
        console.error("[EAFC Grinder] 填充阵容时出错:", error);
        alert("❌ 填充失败：" + error.message);
        fillBtn.disabled = false;
        fillBtn.textContent = "⚡ 一键填充最优阵容";
        fillBtn.style.opacity = "1";
      }
    },

    initSBCQuickNav() {
      const self = this;
      this.loadSBCQuickNav();

      if (typeof UTSBCHubView !== 'undefined' && UTSBCHubView.prototype.populateTiles) {
        const originalPopulateTiles = UTSBCHubView.prototype.populateTiles;
        UTSBCHubView.prototype.populateTiles = function(e, t) {
          originalPopulateTiles.apply(this, arguments);
          setTimeout(() => {
            self._addQuickNavButtonsToSBCHub(this);
          }, 100);
        };
        console.log("[EAFC Grinder] SBC 快捷栏已注册到 UTSBCHubView");
      } else {
        console.warn("[EAFC Grinder] UTSBCHubView 不可用，SBC 快捷栏未注册");
      }
    },

    loadSBCQuickNav() {
      try {
        const raw = GM_getValue("eafc_grinder_sbc_quicknav", "[]");
        const list = JSON.parse(raw);
        if (Array.isArray(list)) {
          this.sbcQuickNavList = list
            .map(item => ({ id: Number(item.id), assetId: item.assetId }))
            .slice(0, this.sbcQuickNavMaxCount);
        }
      } catch (e) {
        this.sbcQuickNavList = [];
      }
    },

    saveSBCQuickNav() {
      GM_setValue("eafc_grinder_sbc_quicknav", JSON.stringify(this.sbcQuickNavList));
    },

    _addQuickNavButtonsToSBCHub(hubView) {
      if (!hubView.sbcSetTiles) return;
      hubView.sbcSetTiles.forEach(tile => {
        const set = tile.data;
        if (!set || !set.id) return;
        const root = tile.__root || (tile.getRootElement && tile.getRootElement());
        if (!root) return;
        if (root.querySelector('.eafc-grinder-quicknav-btn')) return;

        // 需要 tile 的父级有 position 才能让 absolute 定位生效
        if (getComputedStyle(root).position === 'static') {
          root.style.position = 'relative';
        }

        const btn = document.createElement('button');
        btn.className = 'eafc-grinder-quicknav-btn';
        btn.dataset.setId = set.id;
        const isInNav = this.sbcQuickNavList.some(item => item.id === set.id);
        btn.textContent = isInNav ? '从快捷栏去除' : '添加至快捷栏';
        const stopEvt = (e) => { e.stopPropagation(); };
        btn.addEventListener('mousedown', stopEvt);
        btn.addEventListener('touchstart', stopEvt, { passive: false });
        btn.addEventListener('click', (e) => {
          e.stopPropagation();
          e.preventDefault();
          if (this.sbcQuickNavList.some(item => item.id === set.id)) {
            this.removeFromSBCQuickNav(set.id);
            btn.textContent = '添加至快捷栏';
          } else {
            this.addToSBCQuickNav(set.id, set.assetId);
            console.log("[EAFC Grinder] SBC快捷栏 添加 set.id:", JSON.stringify(set.id), "类型:", typeof set.id, "Number:", Number(set.id), "assetId:", set.assetId);
            btn.textContent = '从快捷栏去除';
          }
        });
        root.appendChild(btn);
      });
    },

    addToSBCQuickNav(setId, assetId) {
      const numericId = Number(setId);
      const idx = this.sbcQuickNavList.findIndex(item => item.id === numericId);
      if (idx !== -1) {
        this.sbcQuickNavList.splice(idx, 1);
      }
      this.sbcQuickNavList.unshift({ id: numericId, assetId: assetId });
      if (this.sbcQuickNavList.length > this.sbcQuickNavMaxCount) {
        this.sbcQuickNavList = this.sbcQuickNavList.slice(0, this.sbcQuickNavMaxCount);
      }
      this.saveSBCQuickNav();
      this.updateSBCQuickNavButtons();
      this.renderPanel();
    },

    removeFromSBCQuickNav(setId) {
      const numericId = Number(setId);
      this.sbcQuickNavList = this.sbcQuickNavList.filter(item => item.id !== numericId);
      this.saveSBCQuickNav();
      this.updateSBCQuickNavButtons();
      this.renderPanel();
    },

    updateSBCQuickNavButtons() {
      document.querySelectorAll('.eafc-grinder-quicknav-btn').forEach(btn => {
        const setId = Number(btn.dataset.setId);
        const isInNav = this.sbcQuickNavList.some(item => item.id === setId);
        btn.textContent = isInNav ? '从快捷栏去除' : '添加至快捷栏';
      });
    },

    renderSBCQuickNav() {
      if (!this.sbcQuickNavList.length) return null;

      const container = document.createElement('div');
      container.className = 'grinder-sbc-quicknav';

      const header = document.createElement('div');
      header.className = 'grinder-sbc-quicknav-header';

      const label = document.createElement('span');
      label.textContent = 'SBC 快捷栏';
      header.appendChild(label);

      const deleteToggle = document.createElement('button');
      deleteToggle.className = 'grinder-sbc-quicknav-delete-toggle' + (this.sbcQuickNavDeleteMode ? ' active' : '');
      deleteToggle.textContent = this.sbcQuickNavDeleteMode ? '退出删除' : '删除模式';
      deleteToggle.onclick = () => {
        this.sbcQuickNavDeleteMode = !this.sbcQuickNavDeleteMode;
        this.renderPanel();
      };
      header.appendChild(deleteToggle);
      container.appendChild(header);

      const iconsRow = document.createElement('div');
      iconsRow.className = 'grinder-sbc-quicknav-icons';

      this.sbcQuickNavList.forEach(item => {
        const wrapper = document.createElement('div');
        wrapper.className = 'grinder-sbc-quicknav-item' + (this.sbcQuickNavDeleteMode ? ' delete-mode' : '');
        wrapper.title = '点击' + (this.sbcQuickNavDeleteMode ? '删除' : '跳转');

        const img = document.createElement('img');
        if (typeof AssetLocationUtils !== 'undefined' && AssetLocationUtils.getSquadBuildingSetImageUri) {
          img.src = AssetLocationUtils.getSquadBuildingSetImageUri(item.assetId);
        } else {
          img.src = '';
        }
        img.alt = 'SBC';
        img.onerror = () => { img.style.display = 'none'; };

        wrapper.appendChild(img);
        wrapper.onclick = () => {
          if (this.sbcQuickNavDeleteMode) {
            this.removeFromSBCQuickNav(item.id);
          } else {
            const tryGoToSBC = () => {
              const set = services.SBC.repository.getSetById(Number(item.id));
              if (set) {
                this.goToSBC(set);
              } else {
                console.warn('[EAFC Grinder] 找不到 SBC，尝试加载SBC列表:', item.id);
                const ctrl = _appMain?.getRootViewController?.();
                if (ctrl) {
                  services.SBC.requestSets().observe(ctrl, (e, t) => {
                    e.unobserve(ctrl);
                    if (t.success) {
                      const set2 = services.SBC.repository.getSetById(Number(item.id));
                      if (set2) {
                        this.goToSBC(set2);
                      } else {
                        console.warn('[EAFC Grinder] 加载SBC列表后仍找不到 SBC:', item.id);
                      }
                    } else {
                      console.warn('[EAFC Grinder] 加载SBC列表失败:', t.error || t.status);
                    }
                  });
                }
              }
            };
            tryGoToSBC();
          }
        };
        iconsRow.appendChild(wrapper);
      });

      container.appendChild(iconsRow);
      return container;
    },

    goToSBC(SBCSetEntity) {
      try {
        const controller = _appMain
          ?._rootViewController
          ?.currentController
          ?.currentController
          ?.currentController;
        if (!controller) {
          console.warn('[EAFC Grinder] 无法获取当前控制器');
          return;
        }
        const view = controller.getView ? controller.getView() : null;
        if (view) view.setInteractionState(false);

        services.SBC.requestChallengesForSet(SBCSetEntity).observe(controller, (e, t) => {
          e.unobserve(controller);
          if (t.success && t.data.challenges.length > 0) {
            if (SBCSetEntity.hidden) {
              const r = t.data.challenges[0];
              services.SBC.loadChallenge(r).observe(controller, (ee, tt) => {
                ee.unobserve(controller);
                if (tt.success) {
                  const nav = controller.getNavigationController ? controller.getNavigationController() : null;
                  if (nav) {
                    if (!SBCSetEntity.getChallenge(r.id).squad) {
                      SBCSetEntity.getChallenge(r.id).update(r);
                    }
                    const isMobile = (typeof isPhone === 'function' && isPhone());
                    const o = isMobile ? new UTSBCSquadOverviewViewController() : new UTSBCSquadSplitViewController();
                    o.initWithSBCSet(SBCSetEntity, r.id);
                    nav.pushViewController(o);
                  }
                }
                if (view) view.setInteractionState(true);
              });
            } else {
              const nav = controller.getNavigationController ? controller.getNavigationController() : null;
              if (nav) {
                const isMobile = (typeof isPhone === 'function' && isPhone());
                const o = isMobile ? new UTSBCChallengesViewController() : new UTSBCGroupChallengeSplitViewController();
                o.initWithSBCSet(SBCSetEntity);
                nav.pushViewController(o, true);
                nav.setNavigationTitle(SBCSetEntity.name);
              }
              if (view) view.setInteractionState(true);
            }
          } else {
            console.error('[EAFC Grinder] 获取 SBC challenges 失败:', t);
            if (view) view.setInteractionState(true);
          }
        });
      } catch (err) {
        console.error('[EAFC Grinder] goToSBC 失败:', err);
      }
    },

    parseSBCRequirements(requirements) {
      const reqs = {
        minPlayers: 0,
        minRating: 0,
        maxRating: 999,
        minChemistry: 0,          // key 35: CHEMISTRY_POINTS
        // 稀有度组要求数组：[{groupId, minCount, maxCount}]
        // scope=0 至少, scope=1 至多, scope=2 恰好
        rarityGroups: [],
        // 阵容评分
        minTeamRating: 0,         // key 19: TEAM_RATING - 阵容平均评分下限
        // 全局卡片等级：scope=0 至少, scope=1 至多, scope=2 恰好
        globalQualityMin: 0,      // key 3 scope=0: 至少该等级 (1=铜, 2=银, 3=金)
        globalQualityMax: 0,      // key 3 scope=1: 至多该等级
        // 同X数量：scope=0 至少, scope=1 至多, scope=2 恰好
        sameNationMin: 0,         // key 4 scope=0
        sameNationMax: 0,         // key 4 scope=1/2
        sameLeagueMin: 0,         // key 5 scope=0
        sameLeagueMax: 0,         // key 5 scope=1/2
        sameClubMin: 0,           // key 6 scope=0
        sameClubMax: 0,           // key 6 scope=1/2
        // 不同X数量：scope=0 至少, scope=1 至多, scope=2 恰好
        nationCountMin: 0,        // key 7 scope=0
        nationCountMax: 0,        // key 7 scope=1/2
        leagueCountMin: 0,        // key 8 scope=0
        leagueCountMax: 0,        // key 8 scope=1/2
        clubCountMin: 0,          // key 9 scope=0
        clubCountMax: 0,          // key 9 scope=1/2
        legendCountMin: 0,        // key 15 scope=0
        legendCountMax: 0,        // key 15 scope=1/2
        firstOwnerMin: 0,         // key 30 scope=0
        firstOwnerMax: 0,         // key 30 scope=1/2
        tradability: null,        // key 33: PLAYER_TRADABILITY
        requiredPositions: [],
        requiredLeague: null,
        requiredNation: null,
        requiredClub: null,
        ratingRanges: [],
      };

      if (!requirements || requirements.length === 0) return reqs;

      // 辅助：根据 scope 设置 min/max 值
      // scope=0: 至少 → 设 min; scope=1: 至多 → 设 max; scope=2: 恰好 → 设 min 和 max
      const setByScope = (minField, maxField, value, scope) => {
        if (scope === 0) {
          reqs[minField] = value;
        } else if (scope === 1) {
          reqs[maxField] = value;
        } else if (scope === 2) {
          reqs[minField] = value;
          reqs[maxField] = value;
        }
      };

      requirements.forEach((req) => {
        try {
          const count = req.count;
          const scope = req.scope != null ? req.scope : 0; // 默认 scope=0（至少）
          const collection = req.kvPairs && req.kvPairs._collection ? req.kvPairs._collection : {};
          const keys = Object.keys(collection);

          if (keys.length === 0) return;

          let isGlobal = count === -1;
          const slot = { count: count, min: 0, max: 999 };

          keys.forEach(key => {
            const numKey = parseInt(key, 10);
            const value = collection[key];
            const firstValue = Array.isArray(value) ? value[0] : value;

            switch (numKey) {
              case 2:  // PLAYER_COUNT - 阵容球员总数（全局）
                isGlobal = true;
                break;
              case 3:  // PLAYER_QUALITY 卡片等级 (1=铜, 2=银, 3=金)
                setByScope('globalQualityMin', 'globalQualityMax', firstValue, scope);
                isGlobal = true;
                break;
              case 4:  // SAME_NATION_COUNT
                setByScope('sameNationMin', 'sameNationMax', firstValue, scope);
                isGlobal = true;
                break;
              case 5:  // SAME_LEAGUE_COUNT
                setByScope('sameLeagueMin', 'sameLeagueMax', firstValue, scope);
                isGlobal = true;
                break;
              case 6:  // SAME_CLUB_COUNT
                setByScope('sameClubMin', 'sameClubMax', firstValue, scope);
                isGlobal = true;
                break;
              case 7:  // NATION_COUNT
                setByScope('nationCountMin', 'nationCountMax', firstValue, scope);
                isGlobal = true;
                break;
              case 8:  // LEAGUE_COUNT
                setByScope('leagueCountMin', 'leagueCountMax', firstValue, scope);
                isGlobal = true;
                break;
              case 9:  // CLUB_COUNT
                setByScope('clubCountMin', 'clubCountMax', firstValue, scope);
                isGlobal = true;
                break;
              case 10: // NATION_ID
                slot.nations = Array.isArray(value) ? value : [value];
                slot.scope = scope;
                break;
              case 11: // LEAGUE_ID
                slot.leagues = Array.isArray(value) ? value : [value];
                slot.scope = scope;
                break;
              case 12: // CLUB_ID
                slot.clubs = Array.isArray(value) ? value : [value];
                slot.scope = scope;
                break;
              case 13: // SCOPE（全局标记）
                isGlobal = true;
                break;
              case 15: // LEGEND_COUNT
                setByScope('legendCountMin', 'legendCountMax', firstValue, scope);
                isGlobal = true;
                break;
              case 17: // PLAYER_LEVEL / PLAYER_QUALITY (1=铜, 2=银, 3=金)
                slot.quality = firstValue;
                slot.scope = scope;
                break;
              case 18: // PLAYER_RARITY - 精确匹配单一稀有度
                slot.exactRare = firstValue;
                break;
              case 19: // TEAM_RATING 阵容平均评分下限
                reqs.minTeamRating = firstValue || 0;
                isGlobal = true;
                break;
              case 25: // PLAYER_RARITY_GROUP: rarityGroup === firstValue 的球员
                {
                  const entry = { groupId: firstValue, minCount: 0, maxCount: 0 };
                  if (scope === 0) {
                    entry.minCount = count > 0 ? count : 1;
                  } else if (scope === 1) {
                    entry.maxCount = count > 0 ? count : 1;
                  } else if (scope === 2) {
                    entry.minCount = count > 0 ? count : 1;
                    entry.maxCount = count > 0 ? count : 1;
                  }
                  // 如果已有同 groupId 的条目，合并 minCount/maxCount
                  const existing = reqs.rarityGroups.find(g => g.groupId === firstValue);
                  if (existing) {
                    if (entry.minCount > 0) existing.minCount = entry.minCount;
                    if (entry.maxCount > 0) existing.maxCount = entry.maxCount;
                  } else {
                    reqs.rarityGroups.push(entry);
                  }
                  isGlobal = true;
                }
                break;
              case 26: // PLAYER_MIN_OVR
                slot.min = firstValue;
                break;
              case 27: // PLAYER_EXACT_OVR
                slot.min = firstValue;
                slot.max = firstValue;
                break;
              case 28: // PLAYER_MAX_OVR
                slot.max = firstValue;
                break;
              case 30: // FIRST_OWNER_PLAYERS_COUNT
                setByScope('firstOwnerMin', 'firstOwnerMax', firstValue, scope);
                isGlobal = true;
                break;
              case 33: // PLAYER_TRADABILITY
                reqs.tradability = firstValue;
                isGlobal = true;
                break;
              case 35: // CHEMISTRY_POINTS
                reqs.minChemistry = firstValue || 0;
                isGlobal = true;
                break;
              case 36: // ALL_PLAYERS_CHEMISTRY_POINTS（暂不处理）
                isGlobal = true;
                break;
            }
          });

          if (!isGlobal && slot.count > 0) {
            reqs.ratingRanges.push(slot);
          }
        } catch (e) {
          // 解析单条需求失败，跳过
        }
      });
      // 可读化输出 SBC 需求
      {
        const lines = ["[EAFC Grinder] SBC 需求解析结果:"];
        if (reqs.minPlayers > 0) lines.push(`  球员总数: ${reqs.minPlayers}`);
        if (reqs.minTeamRating > 0) lines.push(`  阵容评分: ≥${reqs.minTeamRating}`);
        if (reqs.minChemistry > 0) lines.push(`  化学点数: ≥${reqs.minChemistry}`);
        const qualityName = { 1: "铜", 2: "银", 3: "金" };
        if (reqs.globalQualityMin > 0) lines.push(`  卡片等级: 至少${qualityName[reqs.globalQualityMin] || reqs.globalQualityMin}`);
        if (reqs.globalQualityMax > 0) lines.push(`  卡片等级: 至多${qualityName[reqs.globalQualityMax] || reqs.globalQualityMax}`);
        if (reqs.minRating > 0) lines.push(`  最低评分: ${reqs.minRating}`);
        if (reqs.maxRating < 999) lines.push(`  最高评分: ${reqs.maxRating}`);
        for (const rg of reqs.rarityGroups) {
          let groupName = `groupId=${rg.groupId}`;
          try { groupName = services.Localization.localize("Player_Group_" + rg.groupId) || groupName; } catch (e) {}
          if (rg.minCount > 0 && rg.maxCount > 0 && rg.minCount === rg.maxCount) {
            lines.push(`  稀有度组[${groupName}]: 恰好${rg.minCount}名`);
          } else {
            if (rg.minCount > 0) lines.push(`  稀有度组[${groupName}]: 至少${rg.minCount}名`);
            if (rg.maxCount > 0) lines.push(`  稀有度组[${groupName}]: 至多${rg.maxCount}名`);
          }
        }
        if (reqs.sameNationMin > 0) lines.push(`  同国籍: 至少${reqs.sameNationMin}名`);
        if (reqs.sameNationMax > 0) lines.push(`  同国籍: 至多${reqs.sameNationMax}名`);
        if (reqs.sameLeagueMin > 0) lines.push(`  同联赛: 至少${reqs.sameLeagueMin}名`);
        if (reqs.sameLeagueMax > 0) lines.push(`  同联赛: 至多${reqs.sameLeagueMax}名`);
        if (reqs.sameClubMin > 0) lines.push(`  同俱乐部: 至少${reqs.sameClubMin}名`);
        if (reqs.sameClubMax > 0) lines.push(`  同俱乐部: 至多${reqs.sameClubMax}名`);
        if (reqs.nationCountMin > 0) lines.push(`  不同国籍: 至少${reqs.nationCountMin}个`);
        if (reqs.nationCountMax > 0) lines.push(`  不同国籍: 至多${reqs.nationCountMax}个`);
        if (reqs.leagueCountMin > 0) lines.push(`  不同联赛: 至少${reqs.leagueCountMin}个`);
        if (reqs.leagueCountMax > 0) lines.push(`  不同联赛: 至多${reqs.leagueCountMax}个`);
        if (reqs.clubCountMin > 0) lines.push(`  不同俱乐部: 至少${reqs.clubCountMin}个`);
        if (reqs.clubCountMax > 0) lines.push(`  不同俱乐部: 至多${reqs.clubCountMax}个`);
        if (reqs.legendCountMin > 0) lines.push(`  传奇球员: 至少${reqs.legendCountMin}名`);
        if (reqs.legendCountMax > 0) lines.push(`  传奇球员: 至多${reqs.legendCountMax}名`);
        if (reqs.firstOwnerMin > 0) lines.push(`  首次拥有: 至少${reqs.firstOwnerMin}名`);
        if (reqs.firstOwnerMax > 0) lines.push(`  首次拥有: 至多${reqs.firstOwnerMax}名`);
        if (reqs.tradability !== null) lines.push(`  可交易: ${reqs.tradability ? '是' : '否'}`);
        for (const range of reqs.ratingRanges) {
          const parts = [];
          if (range.quality) parts.push(qualityName[range.quality] || `品质${range.quality}`);
          if (range.min > 0 && range.max < 999) parts.push(`评分${range.min}-${range.max}`);
          else if (range.min > 0) parts.push(`评分≥${range.min}`);
          else if (range.max < 999) parts.push(`评分≤${range.max}`);
          if (range.exactRare != null) {
            let rareName = `稀有度${range.exactRare}`;
            try { rareName = services.Localization.localize("item.raretype" + range.exactRare); } catch (e) {}
            parts.push(rareName);
          }
          if (range.leagues) parts.push(range.leagues.map(id => this.getLeagueName(id)).join("/"));
          if (range.nations) parts.push(range.nations.map(id => this.getNationName(id)).join("/"));
          if (range.clubs) parts.push(range.clubs.map(id => `俱乐部${id}`).join("/"));
          const scopeLabel = range.scope === 1 ? "至多" : range.scope === 2 ? "恰好" : "至少";
          lines.push(`  槽位[${parts.join(" + ") || "无限制"}]: ${scopeLabel}${range.count}名`);
        }
        console.log(lines.join("\n"));
      }
      console.log("[EAFC Grinder] SBC 需求(原始):", reqs);
      return reqs;
    },

    // 获取 SBCEligibilityKey 名称
    _eligibilityKeyName(key) {
      if (typeof SBCEligibilityKey === 'undefined') return 'unknown';
      for (const name in SBCEligibilityKey) {
        if (SBCEligibilityKey[name] === key) {
          return name;
        }
      }
      return 'unknown';
    },

    filterEligiblePlayers(requirements) {
      // 合并仓库和俱乐部的球员（仓库在前，同 metaDataId 时仓库卡优先被选中）
      let eligible = [...this.getStoragePlayers(), ...this.getClubPlayers()];

      // 去重（按 id）
      const seenIds = new Set();
      eligible = eligible.filter(p => {
        if (seenIds.has(p.id)) return false;
        seenIds.add(p.id);
        return true;
      });

      // 过滤租借球员
      eligible = eligible.filter(p => !this.isLoanPlayer(p));

      // 过滤进化卡
      eligible = eligible.filter(p => !this.isUpgradedCard(p));

      // 应用全局卡片等级约束（scope=0 至少, scope=1 至多）
      const qualityRatingMap = { 1: [0, 64], 2: [65, 74], 3: [75, 99] };
      if (requirements.globalQualityMin || requirements.globalQualityMax) {
        eligible = eligible.filter(p => {
          if (requirements.globalQualityMin) {
            // 至少该等级 → 评分 >= 该等级最低评分
            const [minR] = qualityRatingMap[requirements.globalQualityMin] || [0, 99];
            if (p.rating < minR) return false;
          }
          if (requirements.globalQualityMax) {
            // 至多该等级 → 评分 <= 该等级最高评分
            const [, maxR] = qualityRatingMap[requirements.globalQualityMax] || [0, 99];
            if (p.rating > maxR) return false;
          }
          return true;
        });
      }

      // 将 scope=0/2 且 count === minPlayers 的 ratingRanges 条件提升为硬过滤
      // scope=0 (至少): 所有球员都必须满足（如"至少11名来自某联赛"=全部来自该联赛）
      // scope=1 (至多): 不能作为硬过滤（如"至多3名同俱乐部"不是所有球员的约束）
      // scope=2 (恰好): 同 scope=0，所有球员都必须满足
      if (requirements.ratingRanges && requirements.minPlayers > 0) {
        for (const range of requirements.ratingRanges) {
          if (range.count !== requirements.minPlayers) continue;
          if (range.scope === 1) continue; // scope=1 (至多) 不提升为硬过滤
          // 该条件每个空位都必须满足，提升为硬约束
          eligible = eligible.filter(p => {
            let minR = range.min || 0;
            let maxR = range.max || 999;
            if (range.quality) {
              if (range.scope === 0) {
                // 至少该品质 → 评分 >= 该品质最低分
                if (range.quality === 1) minR = 0;
                else if (range.quality === 2) minR = 65;
                else if (range.quality === 3) minR = 75;
              } else {
                // scope=2 恰好，或默认行为
                if (range.quality === 1) { minR = 0; maxR = 64; }
                else if (range.quality === 2) { minR = 65; maxR = 74; }
                else if (range.quality === 3) { minR = 75; maxR = 99; }
              }
            }
            if (p.rating < minR || p.rating > maxR) return false;
            if (range.exactRare != null && p.rareflag !== range.exactRare) return false;
            if (range.leagues && !range.leagues.includes(p.leagueId)) return false;
            if (range.nations && !range.nations.includes(p.nationId)) return false;
            if (range.clubs && !range.clubs.includes(p.teamId)) return false;
            return true;
          });
        }
      }

      // 如果没有评分范围需求，使用简单的全局过滤
      if (requirements.ratingRanges.length === 0) {
        if (requirements.requiredLeague) {
          eligible = eligible.filter(p => p.leagueId === requirements.requiredLeague);
        }
        if (requirements.requiredNation) {
          eligible = eligible.filter(p => p.nationId === requirements.requiredNation);
        }
        if (requirements.requiredClub) {
          eligible = eligible.filter(p => p.teamId === requirements.requiredClub);
        }
        eligible = eligible.filter(p => p.rating >= requirements.minRating && p.rating <= requirements.maxRating);
        eligible.sort((a, b) => a.rating - b.rating);
        return eligible;
      }

      eligible.sort((a, b) => a.rating - b.rating);
      return eligible;
    },

    selectOptimalSquad(eligiblePlayers, requirements, fillRatings) {
      const squad = [];
      const usedPlayerIds = new Set();
      const usedMetaDataIds = new Set();

      const totalNeeded = requirements.minPlayers || 11;

      // 排序优先级：不可交易 > common(0) > rare(1) > 特殊卡 > 83以下重复仓库球员优先 > 价格低 > 重复仓库球员优先 > 评分低的优先
      const sortedPlayers = [...eligiblePlayers].sort((a, b) => {
        if (a.tradable !== b.tradable) return a.tradable ? 1 : -1;
        if (a.rareflag !== b.rareflag) return a.rareflag - b.rareflag;
        if (a.isStorage !== b.isStorage) {
		if (a.isStorage && a.rating <= 83) return -1;
		if (b.isStorage && b.rating <= 83) return 1;
	}
        const valA = this.getPlayerPrice(a);
        const valB = this.getPlayerPrice(b);
        if (valA !== valB) return valA - valB;
	if (a.isStorage !== b.isStorage) return a.isStorage ? -1 : 1;
        return a.rating - b.rating;
      });

      const canPick = (p) => {
        if (usedPlayerIds.has(p.id)) return false;
        const mid = p._item?._metaData?.id;
        if (mid != null && usedMetaDataIds.has(mid)) return false;
        return true;
      };

      const pick = (p) => {
        squad.push(p);
        usedPlayerIds.add(p.id);
        const mid = p._item?._metaData?.id;
        if (mid != null) usedMetaDataIds.add(mid);
      };

      // 第零步：选 calculateFillRatings 指定的 slot 卡（如果有）
      if (fillRatings && fillRatings._slotPlayerIds && fillRatings._slotPlayerIds.length > 0) {
        for (const playerId of fillRatings._slotPlayerIds) {
          const p = sortedPlayers.find(sp => sp.id === playerId);
          if (p && canPick(p)) pick(p);
        }
      }

      // 第一步：满足 rarityGroups 需求
      for (const rg of requirements.rarityGroups) {
        if (rg.minCount <= 0) continue;
        const isInGroup = (p) => Array.isArray(p.groups) && p.groups.includes(rg.groupId);

        // 如果 calculateFillRatings 已经指定了最优的特殊卡，优先选它们
        const preferredDefIds = fillRatings && fillRatings._groupPlayerDefIds
          ? new Set(fillRatings._groupPlayerDefIds)
          : null;

        let picked = 0;

        // 先选指定的
        if (preferredDefIds) {
          for (const p of sortedPlayers) {
            if (picked >= rg.minCount) break;
            if (!canPick(p)) continue;
            if (!isInGroup(p)) continue;
            if (!preferredDefIds.has(p.definitionId)) continue;
            pick(p);
            picked++;
          }
        }

        // 不足时再从剩余满足 group 的里补
        for (const p of sortedPlayers) {
          if (picked >= rg.minCount) break;
          if (!canPick(p)) continue;
          if (!isInGroup(p)) continue;
          pick(p);
          picked++;
        }
        if (picked < rg.minCount) {
          throw new Error("库存中找不到足够满足稀有度组(groupId=" + rg.groupId + ")要求的球员，需要 " + rg.minCount + " 张");
        }
      }

      // 第二步：满足 ratingRanges 槽位需求
      // 如果 calculateFillRatings 已经指定了 slot 卡，跳过此步
      const slotIdsFilled = fillRatings && fillRatings._slotPlayerIds ? fillRatings._slotPlayerIds.length : 0;
      const ratingRangesTotal = requirements.ratingRanges.reduce((sum, r) => sum + r.count, 0);
      if (requirements.ratingRanges.length > 0 && slotIdsFilled < ratingRangesTotal) {
        for (const range of requirements.ratingRanges) {
          // scope=1 (至多) 的槽位需求不在此处处理，在最终验证阶段检查
          if (range.scope === 1) continue;

          let minRating = range.min || 0;
          let maxRating = range.max || 999;

          if (range.quality) {
            const qrm = { 1: [0, 64], 2: [65, 74], 3: [75, 99] };
            if (range.scope === 0) {
              // 至少该品质 → 评分 >= 该品质最低分
              [minRating] = qrm[range.quality] || [0, 99];
            } else {
              // scope=2 恰好
              [minRating, maxRating] = qrm[range.quality] || [0, 99];
            }
          }

          let picked = 0;
          // 针对该 range 的候选：优先选价值最低的卡，等价值时仓库优先，再评分低的优先
          const getValue = (p) => this.getPlayerPrice(p);
          const rangeCandidates = sortedPlayers.filter(p => {
            if (p.rating < minRating || p.rating > maxRating) return false;
            if (range.exactRare != null && p.rareflag !== range.exactRare) return false;
            if (range.leagues && !range.leagues.includes(p.leagueId)) return false;
            if (range.nations && !range.nations.includes(p.nationId)) return false;
            if (range.clubs && !range.clubs.includes(p.teamId)) return false;
            // 除非指定稀有度，否则不选特殊卡
            if (range.exactRare == null && p.rareflag > 1) return false;
            return true;
          }).sort((a, b) => {
            const valA = getValue(a);
            const valB = getValue(b);
            if (valA !== valB) return valA - valB;
            if (a.isStorage !== b.isStorage) return a.isStorage ? -1 : 1;
            return a.rating - b.rating;
          });

          for (const p of rangeCandidates) {
            if (picked >= range.count) break;
            if (!canPick(p)) continue;
            pick(p);
            picked++;
          }
          if (picked < range.count) {
            const parts = [];
            if (range.exactRare != null) {
              let rareName = `稀有度${range.exactRare}`;
              try { rareName = services.Localization.localize("item.raretype" + range.exactRare); } catch (e) {}
              parts.push(rareName);
            }
            if (range.leagues) parts.push(range.leagues.map(id => this.getLeagueName(id)).join("/"));
            if (range.nations) parts.push(range.nations.map(id => this.getNationName(id)).join("/"));
            if (range.clubs) parts.push(range.clubs.map(id => `俱乐部${id}`).join("/"));
            if (range.quality) parts.push(range.quality === 1 ? "铜卡" : range.quality === 2 ? "银卡" : range.quality === 3 ? "金卡" : `品质${range.quality}`);
            if (minRating > 0 || maxRating < 99) {
              if (minRating > 0 && maxRating < 99) parts.push(`评分${minRating}-${maxRating}`);
              else if (minRating > 0) parts.push(`评分≥${minRating}`);
              else parts.push(`评分≤${maxRating}`);
            }
            throw new Error("找不到满足要求的球员" + (parts.length > 0 ? ": " + parts.join(" ") : ""));
          }
        }
      }

      // 第三步：按 fillRatings 填充（如果存在 TEAM_RATING 要求）
      let neededRatings = null;
      if (fillRatings && fillRatings.length > 0) {
        neededRatings = {};
        fillRatings.forEach(r => {
          neededRatings[r] = (neededRatings[r] || 0) + 1;
        });
        // 扣除阶段 A 和 B 已选的评分
        squad.forEach(p => {
          const r = p.rating;
          if (neededRatings[r] > 0) {
            neededRatings[r]--;
          }
        });
      }

      // 验证全局数量约束（至少/至多/恰好）
      const validateGlobalCounts = (squadPlayers) => {
        const errors = [];

        // 同国家数量
        const nationGroups = {};
        squadPlayers.forEach(p => { nationGroups[p.nationId] = (nationGroups[p.nationId] || 0) + 1; });
        const maxSameNation = Math.max(...Object.values(nationGroups), 0);
        if (requirements.sameNationMin > 0 && maxSameNation < requirements.sameNationMin) {
          errors.push(`同国家球员至少${requirements.sameNationMin}名，当前最多${maxSameNation}名`);
        }
        if (requirements.sameNationMax > 0 && maxSameNation > requirements.sameNationMax) {
          errors.push(`同国家球员至多${requirements.sameNationMax}名，当前最多${maxSameNation}名`);
        }

        // 同联赛数量
        const leagueGroups = {};
        squadPlayers.forEach(p => { leagueGroups[p.leagueId] = (leagueGroups[p.leagueId] || 0) + 1; });
        const maxSameLeague = Math.max(...Object.values(leagueGroups), 0);
        if (requirements.sameLeagueMin > 0 && maxSameLeague < requirements.sameLeagueMin) {
          errors.push(`同联赛球员至少${requirements.sameLeagueMin}名，当前最多${maxSameLeague}名`);
        }
        if (requirements.sameLeagueMax > 0 && maxSameLeague > requirements.sameLeagueMax) {
          errors.push(`同联赛球员至多${requirements.sameLeagueMax}名，当前最多${maxSameLeague}名`);
        }

        // 同俱乐部数量
        const clubGroups = {};
        squadPlayers.forEach(p => { clubGroups[p.teamId] = (clubGroups[p.teamId] || 0) + 1; });
        const maxSameClub = Math.max(...Object.values(clubGroups), 0);
        if (requirements.sameClubMin > 0 && maxSameClub < requirements.sameClubMin) {
          errors.push(`同俱乐部球员至少${requirements.sameClubMin}名，当前最多${maxSameClub}名`);
        }
        if (requirements.sameClubMax > 0 && maxSameClub > requirements.sameClubMax) {
          errors.push(`同俱乐部球员至多${requirements.sameClubMax}名，当前最多${maxSameClub}名`);
        }

        // 不同国家数量
        const uniqueNations = new Set(squadPlayers.map(p => p.nationId)).size;
        if (requirements.nationCountMin > 0 && uniqueNations < requirements.nationCountMin) {
          errors.push(`至少${requirements.nationCountMin}个不同国家，当前${uniqueNations}个`);
        }
        if (requirements.nationCountMax > 0 && uniqueNations > requirements.nationCountMax) {
          errors.push(`至多${requirements.nationCountMax}个不同国家，当前${uniqueNations}个`);
        }

        // 不同联赛数量
        const uniqueLeagues = new Set(squadPlayers.map(p => p.leagueId)).size;
        if (requirements.leagueCountMin > 0 && uniqueLeagues < requirements.leagueCountMin) {
          errors.push(`至少${requirements.leagueCountMin}个不同联赛，当前${uniqueLeagues}个`);
        }
        if (requirements.leagueCountMax > 0 && uniqueLeagues > requirements.leagueCountMax) {
          errors.push(`至多${requirements.leagueCountMax}个不同联赛，当前${uniqueLeagues}个`);
        }

        // 不同俱乐部数量
        const uniqueClubs = new Set(squadPlayers.map(p => p.teamId)).size;
        if (requirements.clubCountMin > 0 && uniqueClubs < requirements.clubCountMin) {
          errors.push(`至少${requirements.clubCountMin}个不同俱乐部，当前${uniqueClubs}个`);
        }
        if (requirements.clubCountMax > 0 && uniqueClubs > requirements.clubCountMax) {
          errors.push(`至多${requirements.clubCountMax}个不同俱乐部，当前${uniqueClubs}个`);
        }

        // 传奇数量
        const legendCount = squadPlayers.filter(p => p.leagueId === 2124 || p.nationId === 0).length;
        if (requirements.legendCountMin > 0 && legendCount < requirements.legendCountMin) {
          errors.push(`至少${requirements.legendCountMin}名传奇球员，当前${legendCount}名`);
        }
        if (requirements.legendCountMax > 0 && legendCount > requirements.legendCountMax) {
          errors.push(`至多${requirements.legendCountMax}名传奇球员，当前${legendCount}名`);
        }

        // 稀有度组数量
        for (const rg of requirements.rarityGroups) {
          const groupCount = squadPlayers.filter(p => Array.isArray(p.groups) && p.groups.includes(rg.groupId)).length;
          if (rg.minCount > 0 && groupCount < rg.minCount) {
            errors.push(`稀有度组(groupId=${rg.groupId})球员至少${rg.minCount}名，当前${groupCount}名`);
          }
          if (rg.maxCount > 0 && groupCount > rg.maxCount) {
            errors.push(`稀有度组(groupId=${rg.groupId})球员至多${rg.maxCount}名，当前${groupCount}名`);
          }
        }

        // 首发球员数量
        const firstOwnerCount = squadPlayers.filter(p => p._item && p._item.isFirstOwner).length;
        if (requirements.firstOwnerMin > 0 && firstOwnerCount < requirements.firstOwnerMin) {
          errors.push(`首发球员至少${requirements.firstOwnerMin}名，当前${firstOwnerCount}名`);
        }
        if (requirements.firstOwnerMax > 0 && firstOwnerCount > requirements.firstOwnerMax) {
          errors.push(`首发球员至多${requirements.firstOwnerMax}名，当前${firstOwnerCount}名`);
        }

        return errors;
      };

      while (squad.length < totalNeeded) {
        let found = false;

        if (neededRatings) {
          // 优先从 neededRatings 中还有的评分里选
          const neededRatingList = Object.keys(neededRatings)
            .map(Number)
            .filter(r => neededRatings[r] > 0);

          console.log(`[EAFC Grinder] 还需要评分:`, neededRatingList, `neededRatings:`, neededRatings);

          if (neededRatingList.length > 0) {
            for (const p of sortedPlayers) {
              if (!canPick(p)) continue;
              if (!neededRatingList.includes(p.rating)) continue;
              // 除非 fillRatings 指定，否则不选特殊卡
              if (p.rareflag > 1 && (!fillRatings || !fillRatings._slotPlayerIds || !fillRatings._slotPlayerIds.includes(p.id))) continue;
              console.log(`[EAFC Grinder] 选中 ${p.name} rating=${p.rating} rareflag=${p.rareflag}`);
              pick(p);
              neededRatings[p.rating]--;
              found = true;
              break;
            }
          }
        }

        if (!found) {
          // 无 TEAM_RATING 需求时，选最便宜的非特殊卡
          for (const p of sortedPlayers) {
            if (!canPick(p)) continue;
            if (p.rareflag > 1) continue;
            pick(p);
            found = true;
            break;
          }
        }

        if (!found) {
          throw new Error(`库存中可用球员不足，已选 ${squad.length}/${totalNeeded} 人，无法继续填充`);
        }
      }

      // 最终验证全局数量约束
      const validationErrors = validateGlobalCounts(squad);
      if (validationErrors.length > 0) {
        console.warn("[EAFC Grinder] 阵容约束验证问题:", validationErrors);
        // 不阻止填充，但在控制台警告
      }

      return squad;
    },

    getRequiredPlayerCount(requirements) {
      return requirements.minPlayers || 11;
    },

    // FC 阵容评分计算公式
    calculateTeamRating(ratings) {
      if (!ratings || ratings.length === 0) return 0;
      let sum = ratings.reduce((a, b) => a + b, 0);
      const avg = sum / 11;
      ratings.forEach(r => {
        if (r > avg) {
          sum += (r - avg);
        }
      });
      return Math.floor(Math.round(sum) / 11);
    },

    // ========== 化学计算工具方法 ==========

    // 计算正确位置球员的总化学值
    _calcChemistryForCorrectPlayers(correctPlayers) {
      const leagueCount = {}, nationCount = {}, clubCount = {};
      for (const p of correctPlayers) {
        leagueCount[p.leagueId] = (leagueCount[p.leagueId] || 0) + 1;
        nationCount[p.nationId] = (nationCount[p.nationId] || 0) + 1;
        clubCount[p.teamId] = (clubCount[p.teamId] || 0) + 1;
      }
      const lb = n => n >= 8 ? 3 : n >= 5 ? 2 : n >= 3 ? 1 : 0;
      const nb = n => n >= 8 ? 3 : n >= 5 ? 2 : n >= 2 ? 1 : 0;
      const cb = n => n >= 7 ? 3 : n >= 4 ? 2 : n >= 2 ? 1 : 0;

      let total = 0;
      for (const p of correctPlayers) {
        total += Math.min(lb(leagueCount[p.leagueId]) + nb(nationCount[p.nationId]) + cb(clubCount[p.teamId]), 3);
      }
      return total;
    },

    // 匈牙利算法：找最大二分图匹配（球员 -> 可踢位置）
    _maxMatching(squad, positions) {
      const n = squad.length;
      const m = positions.length;
      const matchPos = new Array(m).fill(-1);

      function dfs(playerIdx, visited) {
        for (let posIdx = 0; posIdx < m; posIdx++) {
          if (visited[posIdx]) continue;
          if (!squad[playerIdx].positions.includes(positions[posIdx])) continue;
          visited[posIdx] = true;
          if (matchPos[posIdx] === -1 || dfs(matchPos[posIdx], visited)) {
            matchPos[posIdx] = playerIdx;
            return true;
          }
        }
        return false;
      }

      for (let i = 0; i < n; i++) {
        const visited = new Array(m).fill(false);
        dfs(i, visited);
      }

      const assignment = new Array(n).fill(null);
      const correctPlayers = [];
      const usedPosIdx = new Set();
      for (let posIdx = 0; posIdx < m; posIdx++) {
        if (matchPos[posIdx] !== -1) {
          assignment[matchPos[posIdx]] = positions[posIdx];
          correctPlayers.push(squad[matchPos[posIdx]]);
          usedPosIdx.add(posIdx);
        }
      }

      const remainingPosIdx = [];
      for (let posIdx = 0; posIdx < m; posIdx++) {
        if (!usedPosIdx.has(posIdx)) remainingPosIdx.push(posIdx);
      }
      let rp = 0;
      for (let i = 0; i < n; i++) {
        if (assignment[i] === null) {
          assignment[i] = positions[remainingPosIdx[rp++]];
        }
      }

      return { assignment, correctPlayers, matchPos };
    },

    // 计算阵容化学值（含位置匹配）
    _calculateSquadChemistry(squad, formationPositions) {
      const { assignment, correctPlayers } = this._maxMatching(squad, formationPositions);
      const chemistry = this._calcChemistryForCorrectPlayers(correctPlayers);
      return { chemistry, assignment, correctPlayers };
    },

    // ========== 贪心策略工厂 ==========

    _makeAffinityStrategy(field) {
      return (players, squad) => {
        if (squad.length === 0) return [...players].sort((a, b) => b.rating - a.rating);
        const counts = {};
        for (const p of squad) counts[p[field]] = (counts[p[field]] || 0) + 1;
        const bestId = Number(Object.entries(counts).sort((a, b) => b[1] - a[1])[0][0]);
        return [...players].sort((a, b) => {
          const aMatch = a[field] === bestId ? 1 : 0;
          const bMatch = b[field] === bestId ? 1 : 0;
          if (aMatch !== bMatch) return bMatch - aMatch;
          return b.rating - a.rating;
        });
      };
    },

    // 仓库球员优先策略：仓库卡优先，同属性时评分低优先
    _makeStoragePriorityStrategy() {
      return (players, squad) => {
        return [...players].sort((a, b) => {
          if (a.isStorage !== b.isStorage) return a.isStorage ? -1 : 1;
          if (a.rating <= 83 !== b.rating <= 83) return a.rating <= 83 ? -1 : 1;
          return a.rating - b.rating;
        });
      };
    },

    // 仓库普金优先策略：仓库普金(rareflag=0, rating>=75)最优先，其次仓库其他，同属性时评分低优先
    _makeStorageCommonGoldStrategy() {
      return (players, squad) => {
        return [...players].sort((a, b) => {
          const aCG = a.isStorage && a.rareflag === 0 && a.rating >= 75 ? 2 : a.isStorage ? 1 : 0;
          const bCG = b.isStorage && b.rareflag === 0 && b.rating >= 75 ? 2 : b.isStorage ? 1 : 0;
          if (aCG !== bCG) return bCG - aCG;
          return a.rating - b.rating;
        });
      };
    },

    // 仓库稀有金优先策略：仓库稀有金(rareflag=1, rating>=75)最优先，其次仓库其他，同属性时评分低优先
    _makeStorageRareGoldStrategy() {
      return (players, squad) => {
        return [...players].sort((a, b) => {
          const aRG = a.isStorage && a.rareflag === 1 && a.rating >= 75 ? 2 : a.isStorage ? 1 : 0;
          const bRG = b.isStorage && b.rareflag === 1 && b.rating >= 75 ? 2 : b.isStorage ? 1 : 0;
          if (aRG !== bRG) return bRG - aRG;
          return a.rating - b.rating;
        });
      };
    },

    // 仓库83及以下稀有金优先策略：仓库稀有金(rareflag=1, 75<=rating<=83)最优先，其次仓库其他，同属性时评分低优先
    _makeStorageRareGold83Strategy() {
      return (players, squad) => {
        return [...players].sort((a, b) => {
          const aRG = a.isStorage && a.rareflag === 1 && a.rating >= 75 && a.rating <= 83 ? 2 : a.isStorage ? 1 : 0;
          const bRG = b.isStorage && b.rareflag === 1 && b.rating >= 75 && b.rating <= 83 ? 2 : b.isStorage ? 1 : 0;
          if (aRG !== bRG) return bRG - aRG;
          return a.rating - b.rating;
        });
      };
    },

    _makeTargetAffinityStrategy(field, targetIds) {
      const idSet = new Set(targetIds);
      return (players, squad) => {
        return [...players].sort((a, b) => {
          const aMatch = idSet.has(a[field]) ? 1 : 0;
          const bMatch = idSet.has(b[field]) ? 1 : 0;
          if (aMatch !== bMatch) return bMatch - aMatch;
          return b.rating - a.rating;
        });
      };
    },

    // 联赛+国籍联合亲和策略：优先选同联赛+同国籍的球员（最大化 lb+nb）
    _makeLeagueNationStrategy(leagueId, nationId) {
      return (players, squad) => {
        return [...players].sort((a, b) => {
          const aMatch = (a.leagueId === leagueId ? 1 : 0) + (a.nationId === nationId ? 1 : 0);
          const bMatch = (b.leagueId === leagueId ? 1 : 0) + (b.nationId === nationId ? 1 : 0);
          if (aMatch !== bMatch) return bMatch - aMatch;
          return b.rating - a.rating;
        });
      };
    },

    _makeDiversityStrategy(field) {
      return (players, squad) => {
        if (squad.length === 0) return [...players].sort((a, b) => b.rating - a.rating);
        const existingIds = new Set(squad.map(p => p[field]));
        return [...players].sort((a, b) => {
          const aNew = existingIds.has(a[field]) ? 0 : 1;
          const bNew = existingIds.has(b[field]) ? 0 : 1;
          if (aNew !== bNew) return bNew - aNew;
          return b.rating - a.rating;
        });
      };
    },

    _makeMaxCapStrategy(field, maxCount) {
      return (players, squad) => {
        if (squad.length === 0) return [...players].sort((a, b) => b.rating - a.rating);
        const counts = {};
        for (const p of squad) counts[p[field]] = (counts[p[field]] || 0) + 1;
        return [...players].sort((a, b) => {
          const aAt = (counts[a[field]] || 0) >= maxCount ? 1 : 0;
          const bAt = (counts[b[field]] || 0) >= maxCount ? 1 : 0;
          if (aAt !== bAt) return aAt - bAt;
          const aRatio = (counts[a[field]] || 0) / maxCount;
          const bRatio = (counts[b[field]] || 0) / maxCount;
          if (Math.abs(aRatio - bRatio) > 0.01) return aRatio - bRatio;
          return b.rating - a.rating;
        });
      };
    },

    _makeRarityGroupStrategy(groupId) {
      return (players, squad) => {
        return [...players].sort((a, b) => {
          const aIn = a.groups && a.groups.includes(groupId) ? 1 : 0;
          const bIn = b.groups && b.groups.includes(groupId) ? 1 : 0;
          if (aIn !== bIn) return bIn - aIn;
          return b.rating - a.rating;
        });
      };
    },

    _makeCompositeStrategy(weights) {
      return (players, squad) => {
        return [...players].sort((a, b) => {
          let scoreA = 0, scoreB = 0;
          for (const w of weights) {
            if (w.targetIds) {
              const idSet = new Set(w.targetIds);
              scoreA += (idSet.has(a[w.field]) ? 1 : 0) * w.weight;
              scoreB += (idSet.has(b[w.field]) ? 1 : 0) * w.weight;
            } else if (w.diversity && squad.length > 0) {
              const existingIds = new Set(squad.map(p => p[w.field]));
              scoreA += (existingIds.has(a[w.field]) ? 0 : 1) * w.weight;
              scoreB += (existingIds.has(b[w.field]) ? 0 : 1) * w.weight;
            } else if (w.rarityGroup) {
              scoreA += (a.groups && a.groups.includes(w.rarityGroup) ? 1 : 0) * w.weight;
              scoreB += (b.groups && b.groups.includes(w.rarityGroup) ? 1 : 0) * w.weight;
            } else if (w.maxCap && squad.length > 0) {
              const counts = {};
              for (const p of squad) counts[p[w.field]] = (counts[p[w.field]] || 0) + 1;
              const aCnt = counts[a[w.field]] || 0;
              const bCnt = counts[b[w.field]] || 0;
              scoreA += (aCnt >= w.maxCap ? -2 : 0) * w.weight;
              scoreB += (bCnt >= w.maxCap ? -2 : 0) * w.weight;
              scoreA += (aCnt >= w.maxCap - 1 && aCnt < w.maxCap ? -0.5 : 0) * w.weight;
              scoreB += (bCnt >= w.maxCap - 1 && bCnt < w.maxCap ? -0.5 : 0) * w.weight;
            }
          }
          if (scoreA !== scoreB) return scoreB - scoreA;
          return b.rating - a.rating;
        });
      };
    },

    // 化学最大化策略：用 lb/nb/cb 断点估算边际化学贡献，同时考虑硬约束
    _makeChemistryMaxStrategy(requirements) {
      const lb = n => n >= 8 ? 3 : n >= 5 ? 2 : n >= 3 ? 1 : 0;
      const nb = n => n >= 8 ? 3 : n >= 5 ? 2 : n >= 2 ? 1 : 0;
      const cb = n => n >= 7 ? 3 : n >= 4 ? 2 : n >= 2 ? 1 : 0;

      // 化学增量：把 group 从 oldCount 人变到 newCount 人，总化学值变化多少
      const groupDelta = (breakpointFn, oldCount, newCount) => {
        return breakpointFn(newCount) * newCount - breakpointFn(oldCount) * oldCount;
      };

      return (players, squad) => {
        if (squad.length === 0) {
          // 首选：选同一俱乐部中评级最高的两人，为 sameClubMin 打基础
          if (requirements.sameClubMin >= 2) {
            const byClub = {};
            for (const p of players) {
              if (!byClub[p.teamId]) byClub[p.teamId] = [];
              byClub[p.teamId].push(p);
            }
            const bestClub = Object.values(byClub)
              .filter(arr => arr.length >= requirements.sameClubMin)
              .sort((a, b) => b[0].rating - a[0].rating)[0];
            if (bestClub) {
              const bestClubIds = new Set(bestClub.map(p => p.id));
              return [...players].sort((a, b) => {
                const aIn = bestClubIds.has(a.id) ? 1 : 0;
                const bIn = bestClubIds.has(b.id) ? 1 : 0;
                if (aIn !== bIn) return bIn - aIn;
                return b.rating - a.rating;
              });
            }
          }
          return [...players].sort((a, b) => b.rating - a.rating);
        }

        // 统计当前阵容
        const leagueCounts = {}, nationCounts = {}, clubCounts = {};
        for (const p of squad) {
          leagueCounts[p.leagueId] = (leagueCounts[p.leagueId] || 0) + 1;
          nationCounts[p.nationId] = (nationCounts[p.nationId] || 0) + 1;
          clubCounts[p.teamId] = (clubCounts[p.teamId] || 0) + 1;
        }
        const currentLeagues = new Set(Object.keys(leagueCounts).map(Number));
        const currentNations = new Set(Object.keys(nationCounts).map(Number));
        const currentClubs = new Set(Object.keys(clubCounts).map(Number));
        const squadSize = requirements.minPlayers || 11;
        const remaining = squadSize - squad.length;

        return [...players].sort((a, b) => {
          let scoreA = 0, scoreB = 0;

          // 1. 边际化学贡献
          scoreA += groupDelta(lb, leagueCounts[a.leagueId] || 0, (leagueCounts[a.leagueId] || 0) + 1);
          scoreB += groupDelta(lb, leagueCounts[b.leagueId] || 0, (leagueCounts[b.leagueId] || 0) + 1);
          scoreA += groupDelta(nb, nationCounts[a.nationId] || 0, (nationCounts[a.nationId] || 0) + 1);
          scoreB += groupDelta(nb, nationCounts[b.nationId] || 0, (nationCounts[b.nationId] || 0) + 1);
          scoreA += groupDelta(cb, clubCounts[a.teamId] || 0, (clubCounts[a.teamId] || 0) + 1);
          scoreB += groupDelta(cb, clubCounts[b.teamId] || 0, (clubCounts[b.teamId] || 0) + 1);

          // 2. 硬约束惩罚：新增去重类别超出上限
          if (requirements.nationCountMax > 0 && !currentNations.has(a.nationId) && currentNations.size >= requirements.nationCountMax) scoreA -= 100;
          if (requirements.nationCountMax > 0 && !currentNations.has(b.nationId) && currentNations.size >= requirements.nationCountMax) scoreB -= 100;
          if (requirements.leagueCountMax > 0 && !currentLeagues.has(a.leagueId) && currentLeagues.size >= requirements.leagueCountMax) scoreA -= 100;
          if (requirements.leagueCountMax > 0 && !currentLeagues.has(b.leagueId) && currentLeagues.size >= requirements.leagueCountMax) scoreB -= 100;
          if (requirements.clubCountMax > 0 && !currentClubs.has(a.teamId) && currentClubs.size >= requirements.clubCountMax) scoreA -= 100;
          if (requirements.clubCountMax > 0 && !currentClubs.has(b.teamId) && currentClubs.size >= requirements.clubCountMax) scoreB -= 100;

          // 3. 硬约束惩罚：同组人数超出上限
          if (requirements.sameClubMax > 0 && (clubCounts[a.teamId] || 0) >= requirements.sameClubMax) scoreA -= 100;
          if (requirements.sameClubMax > 0 && (clubCounts[b.teamId] || 0) >= requirements.sameClubMax) scoreB -= 100;
          if (requirements.sameNationMax > 0 && (nationCounts[a.nationId] || 0) >= requirements.sameNationMax) scoreA -= 100;
          if (requirements.sameNationMax > 0 && (nationCounts[b.nationId] || 0) >= requirements.sameNationMax) scoreB -= 100;
          if (requirements.sameLeagueMax > 0 && (leagueCounts[a.leagueId] || 0) >= requirements.sameLeagueMax) scoreA -= 100;
          if (requirements.sameLeagueMax > 0 && (leagueCounts[b.leagueId] || 0) >= requirements.sameLeagueMax) scoreB -= 100;

          // 4. 奖励满足最低要求的关键节点
          if (requirements.sameClubMin > 0 && (clubCounts[a.teamId] || 0) + 1 >= requirements.sameClubMin && (clubCounts[a.teamId] || 0) < requirements.sameClubMin) scoreA += 15;
          if (requirements.sameClubMin > 0 && (clubCounts[b.teamId] || 0) + 1 >= requirements.sameClubMin && (clubCounts[b.teamId] || 0) < requirements.sameClubMin) scoreB += 15;

          // 5. 去重下限：如果还缺联赛/国籍/俱乐部数，奖励新增
          if (requirements.leagueCountMin > 0 && currentLeagues.size < requirements.leagueCountMin) {
            // 还需引入几个新联赛？
            const leaguesStillNeeded = requirements.leagueCountMin - currentLeagues.size;
            // 剩余位置还够不够引入新联赛？
            if (leaguesStillNeeded >= remaining) {
              // 必须每人都来自新联赛，奖励新联赛
              if (!currentLeagues.has(a.leagueId)) scoreA += 10;
              if (!currentLeagues.has(b.leagueId)) scoreB += 10;
            } else if (leaguesStillNeeded > 0) {
              // 需要引入但不是每人都要新联赛，轻微奖励新联赛
              if (!currentLeagues.has(a.leagueId)) scoreA += 3;
              if (!currentLeagues.has(b.leagueId)) scoreB += 3;
            }
          }
          if (requirements.nationCountMin > 0 && currentNations.size < requirements.nationCountMin) {
            if (!currentNations.has(a.nationId)) scoreA += 3;
            if (!currentNations.has(b.nationId)) scoreB += 3;
          }
          if (requirements.clubCountMin > 0 && currentClubs.size < requirements.clubCountMin) {
            if (!currentClubs.has(a.teamId)) scoreA += 3;
            if (!currentClubs.has(b.teamId)) scoreB += 3;
          }

          if (scoreA !== scoreB) return scoreB - scoreA;
          return b.rating - a.rating;
        });
      };
    },

    // ========== 化学感知阵容求解器 ==========

    // 球员综合成本（用于化学求解器内部排序）
    // 优先级：不可交易 > 价格低 > 低稀有度 > 仓库 > 评分低
    // 仓库83及以下金卡价格打8折
    _chemPrice(p) {
      let price = this.getPlayerPrice(p);
      if (p.isStorage && p.rating >= 75 && p.rating <= 83) price = Math.round(price * 0.8);
      const tradablePenalty = p.tradable ? 1 : 0;     // 不可交易优先
      const rarePenalty = p.rareflag || 0;             // 低稀有度优先
      const storageBonus = p.isStorage ? 0 : 1;        // 仓库优先
      return tradablePenalty * 1e15 + price * 1e7 + rarePenalty * 1e5 + storageBonus * 1e3 + p.rating;
    },

    // 检查球员是否匹配某个 ratingRange 条件
    _playerMatchesRange(p, range) {
      const qualityMap = { 1: [0, 64], 2: [65, 74], 3: [75, 99] };
      let minR = range.min || 0;
      let maxR = range.max || 999;
      if (range.quality) {
        const qrm = qualityMap[range.quality] || [0, 99];
        if (range.scope === 0) {
          minR = Math.max(minR, qrm[0]);
        } else {
          minR = Math.max(minR, qrm[0]);
          maxR = Math.min(maxR, qrm[1]);
        }
      }
      if (p.rating < minR || p.rating > maxR) return false;
      if (range.exactRare != null && p.rareflag !== range.exactRare) return false;
      if (range.leagues && !range.leagues.includes(p.leagueId)) return false;
      if (range.nations && !range.nations.includes(p.nationId)) return false;
      if (range.clubs && !range.clubs.includes(p.teamId)) return false;
      return true;
    },

    // 验证 ratingRanges 槽位约束
    // 关键：一个球员可以同时满足多个 range（如金卡法甲球员既满足法甲范围也满足金卡范围）
    // 每个 range 独立检查：scope=0/2 检查匹配数 >= count，scope=1 检查匹配数 <= count
    _validateRatingRanges(squad, requirements) {
      const ranges = requirements.ratingRanges || [];
      if (ranges.length === 0) return true;

      for (const range of ranges) {
        const needed = range.count || 0;
        if (needed <= 0) continue;
        const scope = range.scope != null ? range.scope : 0;
        const matches = squad.filter(p => this._playerMatchesRange(p, range)).length;
        if (scope === 1) {
          // 至多 N 名
          if (matches > needed) {
            range._failReason = `至多${needed}名但实际${matches}名`;
            return false;
          }
        } else {
          // scope=0 (至少) 或 scope=2 (恰好) — 都检查 >= N
          if (matches < needed) {
            range._failReason = `需${needed}名但匹配${matches}名`;
            return false;
          }
        }
      }
      return true;
    },

    // 综合评估阵容是否满足所有约束
    _evaluateSquad(squad, requirements, formationPositions) {
      const squadSize = requirements.minPlayers || 11;
      if (squad.length !== squadSize) return { valid: false, reason: `人数${squad.length}≠${squadSize}` };

      // 同 metaDataId 不能重复出现在阵容中（同一球员的不同实例）
      const seenMetaIds = new Set();
      for (const p of squad) {
        if (p.metaDataId && seenMetaIds.has(p.metaDataId)) return { valid: false, reason: `metaDataId重复(${p.metaDataId})` };
        seenMetaIds.add(p.metaDataId);
      }

      const { assignment, correctPlayers, matchPos } = this._maxMatching(squad, formationPositions);
      const chemistry = this._calcChemistryForCorrectPlayers(correctPlayers);

      const teamRating = this.calculateTeamRating(squad.map(p => p.rating));
      if (requirements.minTeamRating > 0 && teamRating < requirements.minTeamRating) return { valid: false, reason: `评级${teamRating}<${requirements.minTeamRating}`, chemistry };
      if (requirements.minChemistry > 0 && chemistry < requirements.minChemistry) return { valid: false, reason: `化学${chemistry}<${requirements.minChemistry}`, chemistry };

      // 稀有度组人数检查
      for (const rg of requirements.rarityGroups) {
        const groupCount = squad.filter(p => p.groups && p.groups.includes(rg.groupId)).length;
        if (rg.minCount > 0 && groupCount < rg.minCount) return { valid: false, reason: `稀有度组(groupId=${rg.groupId})${groupCount}<${rg.minCount}`, chemistry };
        if (rg.maxCount > 0 && groupCount > rg.maxCount) return { valid: false, reason: `稀有度组(groupId=${rg.groupId})${groupCount}>${rg.maxCount}`, chemistry };
      }

      // 同俱乐部数量
      const clubGroups = {};
      for (const p of squad) clubGroups[p.teamId] = (clubGroups[p.teamId] || 0) + 1;
      const maxSameClub = Math.max(...Object.values(clubGroups), 0);
      if (requirements.sameClubMin > 0 && maxSameClub < requirements.sameClubMin) return { valid: false, reason: `同俱乐部${maxSameClub}<${requirements.sameClubMin}`, chemistry };
      if (requirements.sameClubMax > 0 && maxSameClub > requirements.sameClubMax) return { valid: false, reason: `同俱乐部${maxSameClub}>${requirements.sameClubMax}`, chemistry };

      // 同国籍数量
      const nationGroups = {};
      for (const p of squad) nationGroups[p.nationId] = (nationGroups[p.nationId] || 0) + 1;
      const maxSameNation = Math.max(...Object.values(nationGroups), 0);
      if (requirements.sameNationMin > 0 && maxSameNation < requirements.sameNationMin) return { valid: false, reason: `同国籍${maxSameNation}<${requirements.sameNationMin}`, chemistry };
      if (requirements.sameNationMax > 0 && maxSameNation > requirements.sameNationMax) return { valid: false, reason: `同国籍${maxSameNation}>${requirements.sameNationMax}`, chemistry };

      // 同联赛数量
      const leagueGroups = {};
      for (const p of squad) leagueGroups[p.leagueId] = (leagueGroups[p.leagueId] || 0) + 1;
      const maxSameLeague = Math.max(...Object.values(leagueGroups), 0);
      if (requirements.sameLeagueMin > 0 && maxSameLeague < requirements.sameLeagueMin) return { valid: false, reason: `同联赛${maxSameLeague}<${requirements.sameLeagueMin}`, chemistry };
      if (requirements.sameLeagueMax > 0 && maxSameLeague > requirements.sameLeagueMax) return { valid: false, reason: `同联赛${maxSameLeague}>${requirements.sameLeagueMax}`, chemistry };

      // 不同国籍数量
      const uniqueNations = new Set(squad.map(p => p.nationId)).size;
      if (requirements.nationCountMin > 0 && uniqueNations < requirements.nationCountMin) return { valid: false, reason: `不同国籍${uniqueNations}<${requirements.nationCountMin}`, chemistry };
      if (requirements.nationCountMax > 0 && uniqueNations > requirements.nationCountMax) return { valid: false, reason: `不同国籍${uniqueNations}>${requirements.nationCountMax}`, chemistry };

      // 不同联赛数量
      const uniqueLeagues = new Set(squad.map(p => p.leagueId)).size;
      if (requirements.leagueCountMin > 0 && uniqueLeagues < requirements.leagueCountMin) return { valid: false, reason: `不同联赛${uniqueLeagues}<${requirements.leagueCountMin}`, chemistry };
      if (requirements.leagueCountMax > 0 && uniqueLeagues > requirements.leagueCountMax) return { valid: false, reason: `不同联赛${uniqueLeagues}>${requirements.leagueCountMax}`, chemistry };

      // 不同俱乐部数量
      const uniqueClubs = new Set(squad.map(p => p.teamId)).size;
      if (requirements.clubCountMin > 0 && uniqueClubs < requirements.clubCountMin) return { valid: false, reason: `不同俱乐部${uniqueClubs}<${requirements.clubCountMin}`, chemistry };
      if (requirements.clubCountMax > 0 && uniqueClubs > requirements.clubCountMax) return { valid: false, reason: `不同俱乐部${uniqueClubs}>${requirements.clubCountMax}`, chemistry };

      // 传奇数量
      const legendCount = squad.filter(p => p.leagueId === 2124 || p.nationId === 0).length;
      if (requirements.legendCountMin > 0 && legendCount < requirements.legendCountMin) return { valid: false, reason: `传奇${legendCount}<${requirements.legendCountMin}`, chemistry };
      if (requirements.legendCountMax > 0 && legendCount > requirements.legendCountMax) return { valid: false, reason: `传奇${legendCount}>${requirements.legendCountMax}`, chemistry };

      // 首发球员数量
      const firstOwnerCount = squad.filter(p => p._item && p._item.isFirstOwner).length;
      if (requirements.firstOwnerMin > 0 && firstOwnerCount < requirements.firstOwnerMin) return { valid: false, reason: `首发${firstOwnerCount}<${requirements.firstOwnerMin}`, chemistry };
      if (requirements.firstOwnerMax > 0 && firstOwnerCount > requirements.firstOwnerMax) return { valid: false, reason: `首发${firstOwnerCount}>${requirements.firstOwnerMax}`, chemistry };

      // ratingRanges 槽位约束
      if (!this._validateRatingRanges(squad, requirements)) {
        const failRange = (requirements.ratingRanges || []).find(r => r._failReason);
        return { valid: false, reason: `槽位约束不满足: ${failRange?._failReason || '未知'}`, chemistry };
      }

      return { valid: true, assignment, correctPlayers, chemistry, matchPos };
    },

    // 化学贡献评分（用于化学亲和策略排序）
    _chemScore(player, squad) {
      let score = 0;
      for (const p of squad) {
        if (p.leagueId === player.leagueId) score += 1;
        if (p.nationId === player.nationId) score += 1;
        if (p.teamId === player.teamId) score += 2;
      }
      return score;
    },

    // 贪心找初始可行解
    _chemGreedyFindSolution(pool, requirements, formationPositions, sortFn, strategyName = '') {
      const squadSize = requirements.minPlayers || 11;
      const usedIds = new Set();
      const squad = [];

      const canPick = (p) => !usedIds.has(p.id);

      // 检查阵容是否违反 cap 约束
      const squadCapViolated = (s) => {
        const countBy = (field) => {
          const g = {};
          for (const p of s) g[p[field]] = (g[p[field]] || 0) + 1;
          return g;
        };
        const maxOf = (field) => Math.max(...Object.values(countBy(field)), 0);
        const distinctOf = (field) => Object.keys(countBy(field)).length;
        if (requirements.sameClubMax > 0 && maxOf('teamId') > requirements.sameClubMax) return true;
        if (requirements.sameNationMax > 0 && maxOf('nationId') > requirements.sameNationMax) return true;
        if (requirements.sameLeagueMax > 0 && maxOf('leagueId') > requirements.sameLeagueMax) return true;
        if (requirements.nationCountMax > 0 && distinctOf('nationId') > requirements.nationCountMax) return true;
        if (requirements.leagueCountMax > 0 && distinctOf('leagueId') > requirements.leagueCountMax) return true;
        if (requirements.clubCountMax > 0 && distinctOf('teamId') > requirements.clubCountMax) return true;
        return false;
      };

      // 硬约束检查：添加该球员后是否违反 cap（用于正常选人流程）
      const wouldViolateCap = (p) => squadCapViolated([...squad, p]);

      // 选人：同 metaDataId 时若新卡更便宜且不违反约束则替换，否则跳过
      const doPick = (p) => {
        if (p.metaDataId) {
          const dupIdx = squad.findIndex(sp => sp.metaDataId === p.metaDataId);
          if (dupIdx >= 0) {
            if (this._chemPrice(p) >= this._chemPrice(squad[dupIdx])) return false;
            const testSquad = [...squad];
            testSquad.splice(dupIdx, 1, p);
            if (squadCapViolated(testSquad)) return false;
            usedIds.delete(squad[dupIdx].id);
            squad.splice(dupIdx, 1);
          }
        }
        squad.push(p);
        usedIds.add(p.id);
        return true;
      };

      // 诊断辅助：输出阵容球员列表
      const diagSquad = () => squad.map(p => `${p.name}(${p.rating})`).join(", ");

      // 第 1 步：优先满足稀有度组约束（优先选与策略目标联赛/国籍一致的球员）
      for (const rg of requirements.rarityGroups) {
        if (rg.minCount <= 0) continue;
        const rarePool = pool.filter(p => canPick(p) && p.groups && p.groups.includes(rg.groupId));
        // 先用 sortFn 排序，再按策略亲和度微调：与策略目标同联赛/国籍的优先
        const sortedRares = sortFn(rarePool, squad);
        const strategyTargetLeagues = sortFn._targetLeagues || [];
        const strategyTargetNations = sortFn._targetNations || [];
        const strategyTargetClubs = sortFn._targetClubs || [];
        if (strategyTargetLeagues.length > 0 || strategyTargetNations.length > 0 || strategyTargetClubs.length > 0) {
          const targetLeagueSet = new Set(strategyTargetLeagues);
          const targetNationSet = new Set(strategyTargetNations);
          const targetClubSet = new Set(strategyTargetClubs);
          sortedRares.sort((a, b) => {
            const aScore = (targetLeagueSet.has(a.leagueId) ? 2 : 0) + (targetNationSet.has(a.nationId) ? 1 : 0) + (targetClubSet.has(a.teamId) ? 2 : 0);
            const bScore = (targetLeagueSet.has(b.leagueId) ? 2 : 0) + (targetNationSet.has(b.nationId) ? 1 : 0) + (targetClubSet.has(b.teamId) ? 2 : 0);
            if (aScore !== bScore) return bScore - aScore;
            return 0; // 保持 sortFn 原顺序
          });
        }
        let picked = 0;
        let skipped = 0;
        for (let i = 0; picked < rg.minCount && i < sortedRares.length; i++) {
          if (!canPick(sortedRares[i])) { skipped++; continue; }
          if (wouldViolateCap(sortedRares[i])) { skipped++; continue; }
          if (doPick(sortedRares[i])) picked++;
          else skipped++;
        }
      }

      // 第 1.5 步：逐个选人满足 ratingRanges 中 scope!=1 的硬性槽位约束
      // 每选一人后重排序，让化学感知策略能基于已选球员做决策
      const hardRanges = (requirements.ratingRanges || []).filter(r => r.scope !== 1);
      const rangeNeeds = hardRanges.map(range => {
        const needed = range.count || 0;
        let minR = range.min || 0;
        let maxR = range.max || 999;
        if (range.quality) {
          const qrm = { 1: [0, 64], 2: [65, 74], 3: [75, 99] };
          if (range.scope === 0) {
            [minR] = qrm[range.quality] || [0, 99];
          } else {
            [minR, maxR] = qrm[range.quality] || [0, 99];
          }
        }
        return { range, needed, minR, maxR, stillNeed: 0 };
      });

      const playerMatchesRange = (p, r) => {
        if (p.rating < r.minR || p.rating > r.maxR) return false;
        if (r.range.exactRare != null && p.rareflag !== r.range.exactRare) return false;
        if (r.range.leagues && !r.range.leagues.includes(p.leagueId)) return false;
        if (r.range.nations && !r.range.nations.includes(p.nationId)) return false;
        if (r.range.clubs && !r.range.clubs.includes(p.teamId)) return false;
        return true;
      };

      // 扣除已在阵容中满足该 range 的球员
      for (const rn of rangeNeeds) {
        const currentMatches = squad.filter(p => playerMatchesRange(p, rn)).length;
        rn.stillNeed = Math.max(0, rn.needed - currentMatches);
      }
      let totalNeed = rangeNeeds.reduce((s, r) => s + r.stillNeed, 0);

      while (totalNeed > 0 && squad.length < squadSize) {
        // 找还缺人最多的 range
        const mostNeeded = rangeNeeds.filter(r => r.stillNeed > 0)
          .sort((a, b) => b.stillNeed - a.stillNeed)[0];
        if (!mostNeeded) break;

        const candidates = pool.filter(p => {
          if (!canPick(p)) return false;
          if (!playerMatchesRange(p, mostNeeded)) return false;
          if (wouldViolateCap(p)) return false;
          return true;
        });
        if (candidates.length === 0) break;

        const sorted = sortFn(candidates, squad);
        if (!doPick(sorted[0])) break;
        mostNeeded.stillNeed--;
        totalNeed--;

        // 新选的球员可能也满足其他 range
        for (const rn of rangeNeeds) {
          if (rn === mostNeeded || rn.stillNeed <= 0) continue;
          if (playerMatchesRange(sorted[0], rn)) {
            rn.stillNeed--;
            totalNeed--;
          }
        }
      }
      // 第 2 步：贪心填充剩余位置
      while (squad.length < squadSize) {
        const remaining = pool.filter(p => canPick(p) && !wouldViolateCap(p));
        if (remaining.length === 0) break;

        const sorted = sortFn(remaining, squad);
        if (!doPick(sorted[0])) break;
      }
      const ev = this._evaluateSquad(squad, requirements, formationPositions);
      if (!ev.valid) {
        const nameTag = strategyName ? `[${strategyName}] ` : '';
        console.log(`[EAFC Grinder] 贪心结果无效: ${nameTag}${ev.reason || '未知'}`);
      }
      return ev.valid ? { squad, ...ev } : null;
    },

    // 局部搜索降成本
    _chemLocalSearch(squad, requirements, formationPositions) {
      let assignment = null, correctPlayers = null, chemistry = 0;
      let improved = true;
      let iterations = 0;
      const maxIterations = 500;

      while (improved && iterations < maxIterations) {
        improved = false;
        iterations++;

        const currentPrice = squad.reduce((s, p) => s + this._chemPrice(p), 0);
        const squadIds = new Set(squad.map(p => p.id));

        const order = squad.map((p, i) => ({ idx: i, price: this._chemPrice(p) }))
                           .sort((a, b) => b.price - a.price);

        for (const { idx } of order) {
          const candidates = this._chemPool.filter(p => {
            if (squadIds.has(p.id)) return false;
            if (this._chemPrice(p) >= this._chemPrice(squad[idx])) return false;
            return true;
          }).sort((a, b) => this._chemPrice(a) - this._chemPrice(b));

          for (const cand of candidates.slice(0, 30)) {
            const newSquad = [...squad];
            newSquad[idx] = cand;
            const ev = this._evaluateSquad(newSquad, requirements, formationPositions);
            if (ev.valid) {
              const newPrice = newSquad.reduce((s, p) => s + this._chemPrice(p), 0);
              if (newPrice < currentPrice) {
                squad = newSquad;
                assignment = ev.assignment;
                correctPlayers = ev.correctPlayers;
                chemistry = ev.chemistry;
                improved = true;
                break;
              }
            }
          }
          if (improved) break;
        }
      }

      console.log("[EAFC Grinder] 化学求解器-局部搜索迭代次数:", iterations);
      return { squad, assignment, correctPlayers, chemistry };
    },

    // 扰动
    _chemPerturb(squad, k, mode) {
      const n = squad.length;
      const newSquad = [...squad];

      if (mode === 'swap') {
        if (n < 2) return newSquad;
        const i = Math.floor(Math.random() * n);
        const j = (i + 1 + Math.floor(Math.random() * (n - 1))) % n;
        [newSquad[i], newSquad[j]] = [newSquad[j], newSquad[i]];
        return newSquad;
      }

      k = Math.min(k, n);

      const idxPool = Array.from({ length: n }, (_, i) => i);
      for (let i = 0; i < k; i++) {
        const j = i + Math.floor(Math.random() * (n - i));
        [idxPool[i], idxPool[j]] = [idxPool[j], idxPool[i]];
      }

      const squadIds = new Set(squad.map(p => p.id));

      for (let i = 0; i < k; i++) {
        const idx = idxPool[i];
        const oldPrice = this._chemPrice(newSquad[idx]);
        const notInSquad = this._chemPool.filter(p => !squadIds.has(p.id));
        if (notInSquad.length === 0) break;

        const cheaper = notInSquad.filter(p => this._chemPrice(p) < oldPrice);
        let cand;
        if (cheaper.length > 0 && Math.random() < 0.8) {
          cand = cheaper[Math.floor(Math.random() * cheaper.length)];
        } else {
          cand = notInSquad[Math.floor(Math.random() * notInSquad.length)];
        }

        squadIds.delete(newSquad[idx].id);
        squadIds.add(cand.id);
        newSquad[idx] = cand;
      }

      return newSquad;
    },

    // 模拟退火
    _chemSimulatedAnnealing(initialSquad, requirements, formationPositions) {
      let currentSquad = [...initialSquad];
      let ev = this._evaluateSquad(currentSquad, requirements, formationPositions);
      let currentAssignment = ev.assignment;
      let currentCorrectPlayers = ev.correctPlayers;
      let currentChemistry = ev.chemistry;
      let currentPrice = currentSquad.reduce((s, p) => s + this._chemPrice(p), 0);

      let bestSquad = [...currentSquad];
      let bestAssignment = currentAssignment;
      let bestCorrectPlayers = currentCorrectPlayers;
      let bestChemistry = currentChemistry;
      let bestPrice = currentPrice;

      // 自适应初始温度
      let sampleDelta = 0;
      let sampleCount = 0;
      for (let i = 0; i < 100; i++) {
        const mode = Math.random() < 0.7 ? 'replace' : 'swap';
        const k = mode === 'replace' ? (1 + Math.floor(Math.random() * 2)) : 1;
        const newSquad = this._chemPerturb(currentSquad, k, mode);
        const newEv = this._evaluateSquad(newSquad, requirements, formationPositions);
        if (newEv.valid) {
          sampleDelta += Math.abs(newSquad.reduce((s, p) => s + this._chemPrice(p), 0) - currentPrice);
          sampleCount++;
        }
      }
      const avgDelta = sampleCount > 0 ? sampleDelta / sampleCount : currentPrice * 0.03;
      let T = Math.max(avgDelta * 1.5, currentPrice * 0.015, 5000);
      const Tmin = Math.max(avgDelta * 0.005, 50);
      const alpha = 0.9995;
      const maxIter = 10000; // 浏览器环境降低迭代次数

      let accepted = 0;
      let improved = 0;
      let validCount = 0;

      for (let iter = 0; iter < maxIter && T > Tmin; iter++) {
        const mode = Math.random() < 0.7 ? 'replace' : 'swap';
        let k = 1;
        if (mode === 'replace') {
          k = T > avgDelta ? (Math.random() < 0.5 ? 2 : 3) : 1;
        }

        const newSquad = this._chemPerturb(currentSquad, k, mode);
        const newEv = this._evaluateSquad(newSquad, requirements, formationPositions);

        if (newEv.valid) {
          validCount++;
          const newPrice = newSquad.reduce((s, p) => s + this._chemPrice(p), 0);
          const delta = newPrice - currentPrice;

          if (delta < 0 || Math.random() < Math.exp(-delta / T)) {
            currentSquad = newSquad;
            currentAssignment = newEv.assignment;
            currentCorrectPlayers = newEv.correctPlayers;
            currentChemistry = newEv.chemistry;
            currentPrice = newPrice;
            accepted++;

            if (newPrice < bestPrice) {
              bestSquad = [...newSquad];
              bestAssignment = newEv.assignment;
              bestCorrectPlayers = newEv.correctPlayers;
              bestChemistry = newEv.chemistry;
              bestPrice = newPrice;
              improved++;
            }
          }
        }

        T *= alpha;
      }

      console.log("[EAFC Grinder] 化学求解器-SA 迭代:", maxIter, "有效:", validCount, "接受:", accepted, "改进:", improved, "最优:", bestPrice.toLocaleString());
      return { squad: bestSquad, assignment: bestAssignment, correctPlayers: bestCorrectPlayers, chemistry: bestChemistry };
    },

        for (const p of leaguePlayers) nationInLeague[p.nationId] = (nationInLeague[p.nationId] || 0) + 1;
        const topNation = Object.entries(nationInLeague).sort((a, b) => b[1] - a[1])[0];
        if (topNation && Number(topNation[1]) >= 2) {
          const nationId = Number(topNation[0]);
          const key = `leagueNation_${leagueId}_${nationId}`;
          dynamicStrategies[key] = this._makeLeagueNationStrategy(leagueId, nationId);
          dynamicStrategies[key]._targetLeagues = [leagueId];
          dynamicStrategies[key]._targetNations = [nationId];
          dynamicStrategyNames.push(key);
          leagueNationPairs.push(`${leagueId}+${nationId}(${topNation[1]}人)`);
        }
      }
      console.log("[EAFC Grinder] 化学求解器-联赛国籍联合:", leagueNationPairs.join(', '));

      Object.assign(greedyStrategies, dynamicStrategies);
      console.log("[EAFC Grinder] 化学求解器-动态策略:", dynamicStrategyNames);

      // 让浏览器渲染进度条的辅助
      const pause = () => new Promise(r => setTimeout(r, 0));

      // 执行贪心搜索
      const strategies = [...dynamicStrategyNames, 'rating', 'chemistry'];
      const initialSolutions = [];

      for (let si = 0; si < strategies.length; si++) {
        const name = strategies[si];
        if (this._chemProgress) { this._chemProgress(Math.round(si / strategies.length * 10)); await pause(); }
        const sol = this._chemGreedyFindSolution(eligiblePlayers, requirements, formationPositions, greedyStrategies[name], name);
        if (sol) {
          const cost = sol.squad.reduce((s, p) => s + this._chemPrice(p), 0);
          console.log(`[EAFC Grinder] 化学求解器-[${name}] 找到初始解, 成本:`, cost.toLocaleString(), "化学:", sol.chemistry);
          initialSolutions.push({ name, ...sol });
        } else {
          console.log(`[EAFC Grinder] 化学求解器-[${name}] 未找到初始解`);
        }
      }

      // 随机补充：如果可行解不足3个，用随机策略补到3个
      const targetSolutions = 3;
      if (initialSolutions.length < targetSolutions) {
        const needMore = targetSolutions - initialSolutions.length;
        console.log(`[EAFC Grinder] 化学求解器-可行解不足${targetSolutions}个(当前${initialSolutions.length})，启动随机补充...`);
        const fallbackTotal = 500;
        let found = 0;
        for (let i = 0; i < fallbackTotal && found < needMore; i++) {
          if (this._chemProgress) { this._chemProgress(10 + Math.round(i / fallbackTotal * 10)); await pause(); }
          const sol = this._chemGreedyFindSolution(eligiblePlayers, requirements, formationPositions, greedyStrategies.random, `random_${i + 1}`);
          if (sol) {
            const cost = sol.squad.reduce((s, p) => s + this._chemPrice(p), 0);
            console.log(`[EAFC Grinder] 化学求解器-[random_${i + 1}] 找到初始解, 成本:`, cost.toLocaleString());
            initialSolutions.push({ name: `random_${i + 1}`, ...sol });
            found++;
          }
        }
      }

      if (initialSolutions.length === 0) {
        throw new Error("化学感知求解器无法找到满足所有约束的阵容");
      }

      // 按成本排序，只取最低的 targetSolutions 个做后续优化
      initialSolutions.sort((a, b) => a.squad.reduce((s, p) => s + this._chemPrice(p), 0) - b.squad.reduce((s, p) => s + this._chemPrice(p), 0));
      const optimizeCandidates = initialSolutions.slice(0, targetSolutions);
      console.log("[EAFC Grinder] 化学求解器-初始解数:", initialSolutions.length, "取前", optimizeCandidates.length, "个优化");

      // 对每个初始解跑 localSearch + SA
      let globalBest = null;
      const totalRuns = optimizeCandidates.length; // each: LS + 2 SA = 3 sub-steps
      let step = 0;
      const totalSteps = totalRuns * 3;
      for (let ii = 0; ii < optimizeCandidates.length; ii++) {
        const initial = optimizeCandidates[ii];
        step++;
        if (this._chemProgress) { this._chemProgress(20 + Math.round(step / totalSteps * 80)); await pause(); }
        const lsResult = this._chemLocalSearch(initial.squad, requirements, formationPositions);
        const lsCost = lsResult.squad.reduce((s, p) => s + this._chemPrice(p), 0);
        console.log("[EAFC Grinder] 化学求解器-局部搜索后成本:", lsCost.toLocaleString());

        let bestSa = lsResult;
        for (let run = 0; run < 2; run++) {
          step++;
          if (this._chemProgress) { this._chemProgress(20 + Math.round(step / totalSteps * 80)); await pause(); }
          const saResult = this._chemSimulatedAnnealing(lsResult.squad, requirements, formationPositions);
          if (saResult.squad.reduce((s, p) => s + this._chemPrice(p), 0) < bestSa.squad.reduce((s, p) => s + this._chemPrice(p), 0)) {
            bestSa = saResult;
          }
        }
        const saCost = bestSa.squad.reduce((s, p) => s + this._chemPrice(p), 0);
        console.log("[EAFC Grinder] 化学求解器-SA后最优成本:", saCost.toLocaleString());

        if (!globalBest || saCost < globalBest.squad.reduce((s, p) => s + this._chemPrice(p), 0)) {
          globalBest = bestSa;
        }
      }

      // 按 assignment 重排阵容：assignment[playerIdx] = positionValue，映射到 formationPositions 索引
      const { assignment } = this._maxMatching(globalBest.squad, formationPositions);
      const finalSquad = new Array(formationPositions.length).fill(null);
      const squad = globalBest.squad;
      const usedSlots = new Set();
      // 第一轮：精确匹配，球员放到对应位置槽
      for (let pi = 0; pi < squad.length; pi++) {
        const posVal = assignment[pi];
        for (let fi = 0; fi < formationPositions.length; fi++) {
          if (usedSlots.has(fi)) continue;
          if (formationPositions[fi] === posVal) {
            finalSquad[fi] = squad[pi];
            usedSlots.add(fi);
            break;
          }
        }
      }
      // 第二轮：剩余球员填空位
      for (let pi = 0; pi < squad.length; pi++) {
        if (usedSlots.size >= formationPositions.length) break;
        if (finalSquad.some(s => s === squad[pi])) continue;
        for (let fi = 0; fi < formationPositions.length; fi++) {
          if (finalSquad[fi] != null) continue;
          finalSquad[fi] = squad[pi];
          break;
        }
      }
      const orderedSquad = finalSquad.filter(p => p != null);

      const finalCost = orderedSquad.reduce((s, p) => s + this._chemPrice(p), 0);
      console.log("[EAFC Grinder] 化学求解器-最终成本:", finalCost.toLocaleString(), "化学:", globalBest.chemistry);

      // 清理临时状态
      delete this._chemPool;

      return orderedSquad;
    },

    // 获取库存评分分布（club + storage）
    getInventoryRatings() {
      const allPlayers = [...this.getClubPlayers(), ...this.getStoragePlayers()];
      const countByRating = {};
      const pricesByRating = {}; // 每个评分下所有卡的有效价格列表，从低到高排序
      const seenIds = new Set();
      const seenMetaIds = new Set();
      allPlayers.forEach(p => {
        if (seenIds.has(p.id)) return;
        const mid = p._item?._metaData?.id;
        if (mid != null && seenMetaIds.has(mid)) return;
        seenIds.add(p.id);
        if (mid != null) seenMetaIds.add(mid);
        const r = p.rating || 0;
        if (r <= 0) return;
        // 特殊卡（rareflag > 1）不纳入普通填充库存统计
        if (p.rareflag > 1) return;
        // 租借卡不纳入普通填充库存统计
        if (this.isLoanPlayer(p)) return;
        countByRating[r] = (countByRating[r] || 0) + 1;
        const cardPrice = this.priceCacheMap[p.definitionId] || 0;
        const minPrice = p._item?._itemPriceLimits?.minimum || 0;
        const effectivePrice = cardPrice > 0 ? cardPrice : minPrice;
        if (!pricesByRating[r]) pricesByRating[r] = [];
        pricesByRating[r].push(effectivePrice);
      });
      // 每个评分的价格列表从低到高排序，选第 N 张时用第 N 便宜的价格
      Object.keys(pricesByRating).forEach(r => {
        pricesByRating[r].sort((a, b) => a - b);
      });
      const uniqueRatings = Object.keys(countByRating)
        .map(Number)
        .sort((a, b) => b - a);
      return { countByRating, uniqueRatings, pricesByRating };
    },

    // 生成有重复组合（非递减序列）
    _multicombinations(arr, k) {
      const result = [];
      const backtrack = (start, current) => {
        if (current.length === k) {
          result.push([...current]);
          return;
        }
        for (let i = start; i < arr.length; i++) {
          current.push(arr[i]);
          backtrack(i, current);
          current.pop();
        }
      };
      backtrack(0, []);
      return result;
    },

    // 获取单个球员的真实价格
    // 获取单个球员的真实价格（无兜底）
    // 只有 priceCacheMap 有值才返回，否则返回 null（显示 N/A）
    getPlayerRealPrice(p) {
      if (this.priceCacheMap[p.definitionId] > 0) {
        return this.priceCacheMap[p.definitionId];
      }
      return null;
    },

    getPlayerPrice(p, pricesByRating) {
      // 使用 priceCacheMap（后端查价结果）
      if (this.priceCacheMap[p.definitionId] > 0) {
        return this.priceCacheMap[p.definitionId];
      }
      // SBC 填充：没有价格则返回最大限制价或 0（表示无法参与成本排序）
      return p._item?._itemPriceLimits?.maximum || 0;
    },

    // 计算满足阵容评分要求的最优填充评分列表
    // rarityGroups 可选，有稀有度组要求时联合优化
    calculateFillRatings(targetRating, controller, rarityGroups, ratingRanges) {
      const cachedCount = Object.values(this.priceCacheMap).filter(v => v > 0).length;
      if (cachedCount === 0) {
        throw new Error("价格数据尚未加载，请稍后再试");
      }

      const squad = controller._squad;
      const brickIndices = squad.getAllBrickIndices ? squad.getAllBrickIndices() : [];
      const totalSlots = 11 - brickIndices.length;

      const inventory = this.getInventoryRatings();
      const haveRatings = inventory.uniqueRatings;
      const haveRatingsCount = inventory.countByRating;
      const pricesByRating = inventory.pricesByRating;
      if (haveRatings.length === 0) throw new Error("库存中没有可用球员");

      const allPlayers = [...this.getClubPlayers(), ...this.getStoragePlayers()];
      const seenIds = new Set();
      const seenMetaIds = new Set();
      let uniquePlayers = allPlayers.filter(p => {
        if (seenIds.has(p.id)) return false;
        const mid = p._item?._metaData?.id;
        if (mid != null && seenMetaIds.has(mid)) return false;
        seenIds.add(p.id);
        if (mid != null) seenMetaIds.add(mid);
        return true;
      });

      // 将 scope=0/2 且 count === totalSlots 的 ratingRanges 条件提升为硬过滤
      if (ratingRanges && ratingRanges.length > 0) {
        for (const range of ratingRanges) {
          if (range.count !== totalSlots) continue;
          if (range.scope === 1) continue; // scope=1 (至多) 不提升为硬过滤
          let minR = range.min || 0;
          let maxR = range.max || 999;
          if (range.quality) {
            const qrm = { 1: [0, 64], 2: [65, 74], 3: [75, 99] };
            if (range.scope === 0) {
              [minR] = qrm[range.quality] || [0, 99];
            } else {
              [minR, maxR] = qrm[range.quality] || [0, 99];
            }
          }
          uniquePlayers = uniquePlayers.filter(p => {
            if (p.rating < minR || p.rating > maxR) return false;
            if (range.exactRare != null && p.rareflag !== range.exactRare) return false;
            if (range.leagues && !range.leagues.includes(p.leagueId)) return false;
            if (range.nations && !range.nations.includes(p.nationId)) return false;
            if (range.clubs && !range.clubs.includes(p.teamId)) return false;
            return true;
          });
        }
      }

      // 辅助：计算纯金卡方案的最优评分组合和总价
      const calcBestNormalCombo = (slots, fixedRatings) => {
        const allRatings = [...fixedRatings, ...Array(slots).fill(0)];

        let basisRating = 0;
        for (let i = haveRatings.length - 1; i >= 0; i--) {
          const r = haveRatings[i];
          const sim = [...fixedRatings, ...Array(slots).fill(r)];
          if (this.calculateTeamRating(sim) >= targetRating) {
            basisRating = r;
            break;
          }
        }
        if (basisRating === 0) return null;

        let fillNumber = 5;
        if (slots <= 3) fillNumber = 9;
        else if (slots === 4) fillNumber = 8;
        else if (slots === 5) fillNumber = 7;
        else if (slots === 6) fillNumber = 6;

        const fillOffset = Math.floor(fillNumber / 2) - 1;
        const basisIndex = haveRatings.indexOf(basisRating);
        let sliceStart = Math.max(0, basisIndex - fillOffset);
        let sliceEnd = Math.min(haveRatings.length, basisIndex + fillNumber - fillOffset);
        const candidates = haveRatings.slice(sliceStart, sliceEnd);

        const combinations = this._multicombinations(candidates, slots);
        let best = null;

        for (const combo of combinations) {
          const sim = [...fixedRatings, ...combo];
          if (this.calculateTeamRating(sim) < targetRating) continue;

          const needCount = {};
          combo.forEach(r => { needCount[r] = (needCount[r] || 0) + 1; });
          let lackCount = 0;
          let totalValue = 0;
          for (const rStr of Object.keys(needCount)) {
            const r = Number(rStr);
            const need = needCount[r];
            const have = haveRatingsCount[r] || 0;
            if (have < need) lackCount += need - have;
            const prices = pricesByRating[r] || [];
            for (let i = 0; i < need; i++) {
              totalValue += prices[i] || 0;
            }
          }

          // 库存不够的组合直接跳过
          if (lackCount > 0) continue;

          if (!best || totalValue < best.totalValue) {
            best = { ratings: combo, totalValue, lackCount };
          }
        }
        return best;
      };

      // === 生成 ratingRanges 的 slot 组合 ===
      let slotPlans = [{ players: [], ratings: [], cost: 0 }];
      if (ratingRanges && ratingRanges.length > 0) {
        for (const range of ratingRanges) {
          // scope=1 (至多) 不作为槽位需求处理
          if (range.scope === 1) continue;

          let minRating = range.min || 0;
          let maxRating = range.max || 999;
          if (range.quality) {
            const qrm = { 1: [0, 64], 2: [65, 74], 3: [75, 99] };
            if (range.scope === 0) {
              [minRating] = qrm[range.quality] || [0, 99];
            } else {
              [minRating, maxRating] = qrm[range.quality] || [0, 99];
            }
          }

          const candidates = uniquePlayers.filter(p => {
            if (p.rating < minRating || p.rating > maxRating) return false;
            if (range.exactRare != null && p.rareflag !== range.exactRare) return false;
            if (range.leagues && !range.leagues.includes(p.leagueId)) return false;
            if (range.nations && !range.nations.includes(p.nationId)) return false;
            if (range.clubs && !range.clubs.includes(p.teamId)) return false;
            return true;
          }).sort((a, b) => {
            return this.getPlayerPrice(a, pricesByRating) - this.getPlayerPrice(b, pricesByRating);
          }).slice(0, 3);

          if (candidates.length < range.count) {
            const parts = [];
            if (range.exactRare != null) {
              let rareName = `稀有度${range.exactRare}`;
              try { rareName = services.Localization.localize("item.raretype" + range.exactRare); } catch (e) {}
              parts.push(rareName);
            }
            if (range.leagues) parts.push(range.leagues.map(id => this.getLeagueName(id)).join("/"));
            if (range.nations) parts.push(range.nations.map(id => this.getNationName(id)).join("/"));
            if (range.clubs) parts.push(range.clubs.map(id => `俱乐部${id}`).join("/"));
            if (range.quality) parts.push(range.quality === 1 ? "铜卡" : range.quality === 2 ? "银卡" : range.quality === 3 ? "金卡" : `品质${range.quality}`);
            if (minRating > 0 || maxRating < 99) {
              if (minRating > 0 && maxRating < 99) parts.push(`评分${minRating}-${maxRating}`);
              else if (minRating > 0) parts.push(`评分≥${minRating}`);
              else parts.push(`评分≤${maxRating}`);
            }
            throw new Error("找不到满足要求的球员" + (parts.length > 0 ? ": " + parts.join(" ") : ""));
          }

          const newPlans = [];
          for (const plan of slotPlans) {
            for (const p of candidates) {
              if (plan.players.some(c => c.id === p.id)) continue;
              const pPrice = this.getPlayerPrice(p, pricesByRating);
              newPlans.push({
                players: [...plan.players, p],
                ratings: [...plan.ratings, p.rating],
                cost: plan.cost + pPrice
              });
            }
          }
          slotPlans = newPlans;
        }
      }

      // === 生成 rarityGroups 的特殊卡组合 ===
      // 对每个有 minCount 要求的 rarityGroup，枚举球员组合，然后叉积合并
      const rgEntries = (rarityGroups || []).filter(g => g.minCount > 0);
      let groupPlans = [{ players: [], ratings: [], cost: 0 }];
      for (const rg of rgEntries) {
        const groupPlayers = uniquePlayers.filter(p =>
          Array.isArray(p.groups) && p.groups.includes(rg.groupId)
        );
        if (groupPlayers.length < rg.minCount) {
          throw new Error("库存中满足稀有度组(groupId=" + rg.groupId + ")要求的球员不足，需要 " + rg.minCount + " 张，只有 " + groupPlayers.length + " 张");
        }
        const seenDef = new Set();
        const uniqueGroupPlayers = groupPlayers.filter(p => {
          if (seenDef.has(p.definitionId)) return false;
          seenDef.add(p.definitionId);
          const price = this.priceCacheMap[p.definitionId];
          if (!price) return false;
          return true;
        });
        const combos = this._multicombinations(uniqueGroupPlayers, rg.minCount);
        const newGroupPlans = [];
        for (const plan of groupPlans) {
          for (const combo of combos) {
            // 避免重复选同一球员
            if (plan.players.some(pp => combo.some(c => c.id === pp.id))) continue;
            newGroupPlans.push({
              players: [...plan.players, ...combo],
              ratings: [...plan.ratings, ...combo.map(p => p.rating)],
              cost: plan.cost + combo.reduce((sum, p) => sum + (this.priceCacheMap[p.definitionId] || 0), 0)
            });
          }
        }
        groupPlans = newGroupPlans;
        if (groupPlans.length === 0) {
          throw new Error("库存中无法同时满足多个稀有度组要求");
        }
      }

      // === 枚举所有组合，找总成本最低且达标的方案 ===
      let bestOverall = null;

      for (const slotPlan of slotPlans) {
        for (const groupPlan of groupPlans) {
          const fixedRatings = [...slotPlan.ratings, ...groupPlan.ratings];
          const remainSlots = totalSlots - slotPlan.players.length - groupPlan.players.length;

          let normalBest = null;
          if (remainSlots > 0) {
            normalBest = calcBestNormalCombo(remainSlots, fixedRatings);
            if (!normalBest) continue;
            if (normalBest.lackCount > 0) continue;
          } else {
            if (this.calculateTeamRating(fixedRatings) < targetRating) continue;
            normalBest = { ratings: [], totalValue: 0, lackCount: 0 };
          }

          const totalCost = slotPlan.cost + groupPlan.cost + normalBest.totalValue;

          if (!bestOverall || totalCost < bestOverall.totalCost) {
            bestOverall = {
              totalCost,
              slotPlan,
              groupPlan,
              normalRatings: normalBest.ratings,
            };
          }
        }
      }

      if (!bestOverall) {
        throw new Error("无法找到满足阵容评分 " + targetRating + " 的方案");
      }

      console.log("[EAFC Grinder] 最优方案, 总成本:", bestOverall.totalCost,
        "slot评分:", bestOverall.slotPlan.ratings,
        "特殊卡评分:", bestOverall.groupPlan.ratings,
        "金卡评分:", bestOverall.normalRatings);

      const result = [
        ...bestOverall.slotPlan.ratings,
        ...bestOverall.groupPlan.ratings,
        ...bestOverall.normalRatings
      ];
      result._slotPlayerIds = [...bestOverall.slotPlan.players, ...bestOverall.groupPlan.players].map(p => p.id);
      result._groupPlayerDefIds = bestOverall.groupPlan.players.map(p => p.definitionId);
      return result;
    },

    async fillSquadWithPlayers(controller, squad) {
      console.log("[EAFC Grinder] 正在将球员填充到阵容...");
      const fillLog = [];

      // 获取锁定的位置（这些位置不需要填充）
      const brickIndices = controller._squad.getAllBrickIndices ? controller._squad.getAllBrickIndices() : [];
      console.log("[EAFC Grinder] 锁定的位置（不需要填充）:", brickIndices);

      // 获取阵容阵型信息
      const formation = controller._challenge.formation;
      const formationData = repositories.Squad.getFormation(formation);
      const generalPositions = formationData.generalPositions;

      // 按照 FSU 的方式构建球员列表
      const playerList = [];
      let squadIndex = 0;

      // 遍历所有位置（0-26），但只填充非锁定的位置
      for (let i = 0; i < generalPositions.length; i++) {
        // 如果是锁定的位置，跳过
        if (brickIndices.includes(i)) {
          console.log(`[EAFC Grinder] 位置 ${i} 是锁定的，跳过`);
          playerList.push(null);
          continue;
        }

        // 如果没有更多球员可填充
        if (squadIndex >= squad.length) {
          console.log(`[EAFC Grinder] 没有更多球员可填充`);
          playerList.push(null);
          continue;
        }

        const player = squad[squadIndex];
        const playerItem = player._item;

        if (playerItem) {
          playerList.push(playerItem);
          fillLog.push(`位置 ${i + 1}: ${player.name} (OVR: ${player.rating}, ${player.position}, mid:${player._item?._metaData?.id ?? "?"})`);
          console.log(`[EAFC Grinder] 位置 ${i} 已分配: ${player.name} (${player.rating})`);
          squadIndex++;
        } else {
          playerList.push(null);
          squadIndex++;
        }
      }

      // 使用 FSU 的方式保存阵容
      try {
        await this.saveSquad(controller._challenge, controller._squad, playerList);
        console.log("[EAFC Grinder] 阵容保存完成");
      } catch (e) {
        console.error("[EAFC Grinder] 保存阵容失败:", e);
        throw e;
      }

      console.log("[EAFC Grinder] 阵容填充完成");
      return fillLog;
    },

    // 保存阵容（参考 FSU 的实现）
    async saveSquad(challenge, squad, playerList) {
      const self = this;

      return new Promise((resolve) => {
        // 移除所有球员
        squad.removeAllItems();

        // 设置球员（此时球员已经显示在阵容里了）
        squad.setPlayers(playerList, true);

        // 设置超时，确保 Promise 一定会 resolve
        const timeoutId = setTimeout(() => {
          console.log("[EAFC Grinder] 保存阵容超时（5秒），但仍然 resolve");
          // 尝试通知变更
          try {
            if (challenge.onDataChange) {
              challenge.onDataChange.notify({ squad: squad });
            }
            // 尝试设置交互状态
            self._setInteractionState();
          } catch (e) {
            console.warn("[EAFC Grinder] 超时处理时出错:", e);
          }
          resolve();
        }, 5000);

        let isResolved = false;
        const doResolve = () => {
          if (!isResolved) {
            isResolved = true;
            clearTimeout(timeoutId);
            console.log("[EAFC Grinder] saveSquad resolve");
            resolve();
          }
        };

        // 尝试保存到服务器
        try {
          console.log("[EAFC Grinder] 开始保存阵容...");
          services.SBC.saveChallenge(challenge).observe(self, (obs, result) => {
            console.log("[EAFC Grinder] saveChallenge 回调触发, success:", result?.success);
            try {
              obs.unobserve(self);

              // 通知数据变更
              if (challenge.onDataChange) {
                challenge.onDataChange.notify({ squad: squad });
                console.log("[EAFC Grinder] 已通知阵容数据变更");
              }

              self._setInteractionState();
              console.log("[EAFC Grinder] 阵容保存流程完成");
            } catch (saveErr) {
              console.warn("[EAFC Grinder] 处理保存结果时出错:", saveErr);
            }
            doResolve();
          });
        } catch (err) {
          console.warn("[EAFC Grinder] saveChallenge 调用异常:", err);
          doResolve();
        }
      });
    },

    // 设置交互状态（和 FSU 一样的方式）
    _setInteractionState() {
      try {
        const appMain = getAppMain ? getAppMain() : _appMain;
        if (appMain && appMain.getRootViewController) {
          const rootVC = appMain.getRootViewController();
          const viewVC = (typeof isPhone !== 'undefined' && isPhone())
            ? rootVC?.currentController
            : rootVC?.leftController;

          if (viewVC && viewVC.getView && viewVC.getView()) {
            const view = viewVC.getView();
            if (view && typeof view.setInteractionState === 'function' && !view._interactionState) {
              view.setInteractionState(true);
              console.log("[EAFC Grinder] 已设置交互状态为 true");
            }
          }
        }
      } catch (e) {
        console.warn("[EAFC Grinder] 设置交互状态失败:", e);
      }
    },
  };

  // 检测登录成功并开始初始化
  function checkAndInit() {
    if (EAFCGrinder.initialized) return;

    try {
      const controllerReady =
        typeof _appMain !== "undefined" &&
        _appMain &&
        _appMain._rootViewController &&
        _appMain._rootViewController.currentController &&
        _appMain._rootViewController.currentController.currentController &&
        _appMain._rootViewController.currentController.currentController.currentController;

      const authReady =
        typeof services !== "undefined" &&
        services.Authentication &&
        services.Authentication.utasSession &&
        services.Authentication.utasSession.id;

      if (controllerReady && authReady) {
        console.log("[EAFC Grinder] 检测到登录成功，开始初始化");
        EAFCGrinder.init();
      }
    } catch (e) {
      // 控制器未就绪，继续等待
    }
  }

  // 使用 MutationObserver 监听 DOM 变化
  const observer = new MutationObserver(() => {
    checkAndInit();
  });

  observer.observe(document.body, {
    childList: true,
    subtree: true,
  });

  // 定期检查
  const checkInterval = setInterval(() => {
    checkAndInit();
    if (EAFCGrinder.initialized) clearInterval(checkInterval);
  }, 1000);
})();
